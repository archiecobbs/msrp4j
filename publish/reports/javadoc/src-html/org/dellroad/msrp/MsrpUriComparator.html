006package org.dellroad.msrp;
007
008import java.net.URI;
009import java.util.Comparator;
010
011/**
012 * Compares MRSP URI's for equality according to RFC 4579, Section 6.1.
013 */
014public final class MsrpUriComparator implements Comparator<MsrpUri> {
015
016    public static final MsrpUriComparator INSTANCE = new MsrpUriComparator();
017
018    private MsrpUriComparator() {
019    }
020
021    @Override
022    public int compare(MsrpUri msrp1, MsrpUri msrp2) {
023        final URI uri1 = msrp1.getUri();
024        final URI uri2 = msrp2.getUri();
025        int diff = uri1.getScheme().toLowerCase().compareTo(uri2.getScheme().toLowerCase());
026        if (diff != 0)
027            return diff;
028        diff = uri1.getHost().toLowerCase().compareTo(uri2.getHost().toLowerCase());
029        if (diff != 0)
030            return diff;
031        diff = Integer.compare(uri1.getPort(), uri2.getPort());
032        if (diff != 0)
033            return diff;
034        diff = msrp1.getSessionId().compareTo(msrp2.getSessionId());
035        if (diff != 0)
036            return diff;
037        diff = msrp1.getTransport().compareTo(msrp2.getTransport());
038        if (diff != 0)
039            return diff;
040        return 0;
041    }
042}
