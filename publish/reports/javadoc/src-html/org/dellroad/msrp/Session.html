088    /**
089     * Close this session.
090     *
091     * <p>
092     * Any {@link FailureListener}s associated with queued messages will be notified.
093     * This method is idempotent.
094     * </p>
095     *
096     * @param cause error that occurred, if any, otherwise null
097     * @return true if this instance was closed, false if this instance was already closed
098     */
099    public boolean close(final Exception cause) {
100        synchronized (this.msrp) {
101            if (this.closed)
102                return false;
103            if (this.log.isDebugEnabled())
104                this.log.debug("closing " + this);
105            try {
106                this.flushOutputQueue();            // to ensure any failure responses get sent
107            } catch (IOException e) {
108                // ignore
109            }
110            for (OutputChunks chunks : this.outputChunks.values()) {
111                chunks.close();
112                if (chunks.hasNext()) {
113                    chunks.notifyFailure(this, this.callbackExecutor,
114                      new Status(MsrpConstants.RESPONSE_CODE_SESSION_DOES_NOT_EXIST, "Session closed"));
115                }
116            }
117            this.inputChunks.clear();
118            this.outputChunks.clear();
119            this.outputTransactions.clear();
120            this.outputQueue.clear();
121            this.closed = true;
122            this.msrp.handleSessionClosed(this);
123
124            // Notify listener
125            this.callbackExecutor.execute(new Runnable() {
126                @Override
127                public void run() {
128                    try {
129                        Session.this.listener.sessionClosed(Session.this, cause);
130                    } catch (ThreadDeath t) {
131                        throw t;
132                    } catch (Throwable t) {
133                        Session.this.log.error("error in listener notification", t);
134                    }
135                }
136            });
137            return true;
138        }
139    }

141    /**
142     * Enqueue an outgoing message specified as a {@code byte[]} array.
143     *
144     * @param content message content
145     * @param contentType content type
146     * @param headers other headers (MIME and/or extension), or null for none
147     * @param reportListener success and/or failure listener, or null for none
148     * @return unique message ID, or null if this instance is closed
149     * @throws IllegalArgumentException if {@code content} or {@code contentType} is null
150     * @throws IllegalArgumentException if {@code headers} contains an invalid header name or value
151     */
152    public String send(byte[] content, String contentType, Iterable<? extends Header> headers, ReportListener reportListener) {
153        if (content == null)
154            throw new IllegalArgumentException("null content");
155        return this.doSend(new ByteArrayInputStream(content), content.length, contentType, headers, reportListener);
156    }

193    /**
194     * Cancel an outgoing message previously sent.
195     *
196     * @param messageId message ID returned by {@code send()}
197     * @throws IllegalArgumentException if {@code messageId} is null
198     * @return true if message was canceled, false if message has already been completely sent
199     */
200    public boolean cancel(String messageId) {
201        if (messageId == null)
202            throw new IllegalArgumentException("null messageId");
203        synchronized (this.msrp) {
204            final OutputChunks chunks = this.outputChunks.get(messageId);
205            if (chunks == null)
206                return false;
207            chunks.close();
208            return chunks.isAborted();
209        }
210    }

212    /**
213     * Enqueue an outgoing success report.
214     *
215     * @param toPath path to the peer; should equal the {@code From-Path} from the received message
216     * @param messageId the {@code Message-ID} from the received message
217     * @param status success status, or null for default {@code 200 Message Delivered}
218     * @param byteRange byte range successfully received
219     * @return true if report was sent, false if report could not be sent because this instance is closed
220     * @throws IllegalArgumentException if {@code toPath} is null or empty
221     * @throws IllegalArgumentException if {@code messageId} is null or invalid
222     * @throws IllegalArgumentException if {@code byteRange} is null
223     */
224    public boolean sendSuccessReport(List<MsrpUri> toPath, String messageId, ByteRange byteRange, Status status) {
225        if (toPath == null || toPath.isEmpty() || toPath.get(0) == null)
226            throw new IllegalArgumentException("null/empty toPath");
227        if (messageId == null)
228            throw new IllegalArgumentException("null messageId");
229        if (byteRange == null)
230            throw new IllegalArgumentException("null byteRange");
231        if (status == null)
232            status = new Status(MsrpConstants.RESPONSE_CODE_OK, "Message delivered");
233        synchronized (this.msrp) {
234            if (this.closed)
235                return false;
236            this.enqueueReport(toPath, messageId, status, byteRange);
237        }
238        return true;
239    }

241    /**
242     * Enqueue an outgoing failure report.
243     *
244     * @param toPath path to the peer; should equal the {@code From-Path} from the received message
245     * @param messageId the {@code Message-ID} from the received message
246     * @param status failure status
247     * @return true if report was sent, false if report could not be sent because this instance is closed
248     * @throws IllegalArgumentException if {@code toPath} is null or empty
249     * @throws IllegalArgumentException if {@code messageId} is null or invalid
250     * @throws IllegalArgumentException if {@code status} is null
251     */
252    public boolean sendFailureReport(List<MsrpUri> toPath, String messageId, Status status) {
253        if (toPath == null || toPath.isEmpty() || toPath.get(0) == null)
254            throw new IllegalArgumentException("null/empty toPath");
255        if (messageId == null)
256            throw new IllegalArgumentException("null messageId");
257        if (status == null)
258            throw new IllegalArgumentException("null status");
259        synchronized (this.msrp) {
260            if (this.closed)
261                return false;
262            this.enqueueReport(toPath, messageId, status, null);
263        }
264        return true;
265    }

269    String doSend(InputStream input, int size, String contentType,
270      Iterable<? extends Header> headers, ReportListener reportListener) {
271        synchronized (this.msrp) {
272
273            // Sanity check
274            if (this.closed)
275                return null;
276
277            // Enqueue output message
278            final OutputChunks chunks = new OutputChunks(this.localURI,
279              this.remoteURI, input, size, contentType, headers, reportListener);
280            final String messageId = chunks.getMessageId();
281            this.outputChunks.put(messageId, chunks);
282
283            // Wakeup MSRP thread so it will invoke performHousekeeping()
284            this.msrp.wakeup();
285
286            // Done
287            return messageId;
288        }
289    }

300    // Handle an incoming {@link MsrpMessage} received on this session.
301    void handleMessage(MsrpMessage msg) {
302
303        // Sanity check
304        if (msg == null)
305            throw new IllegalArgumentException("null msg");
306        if (this.closed)
307            throw new IllegalStateException("session is closed");
308
309        // Handle request or response
310        if (msg instanceof MsrpRequest)
311            this.handleRequest((MsrpRequest)msg);
312        else if (msg instanceof MsrpResponse)
313            this.handleResponse((MsrpResponse)msg);
314        else
315            this.log.error("Session.handleInput(): ignoring unknown message of type " + msg.getClass().getName());
316    }
317
318    // Handle request
319    private void handleRequest(MsrpRequest request) {
320        switch (request.getMethod()) {
321        case MsrpConstants.METHOD_SEND:
322            this.handleSend(request);
323            break;
324        case MsrpConstants.METHOD_REPORT:
325            this.handleReport(request);
326            break;
327        default:
328            if (!FailureReport.NO.equals(request.getHeaders().getFailureReport())) {
329                this.outputQueue.add(Session.createMsrpResponse(request,
330                  MsrpConstants.RESPONSE_CODE_UNKNOWN_METHOD, "Unknown method `" + request.getMethod() + "'"));
331            }
332            return;
333        }
334    }
335
336    // Handle SEND request
337    private void handleSend(MsrpRequest request) {
338
339        // Get/create input message
340        final MsrpHeaders headers = request.getHeaders();
341        final String messageId = headers.getMessageId();
342        InputChunks chunks0 = this.inputChunks.get(messageId);
343        if (chunks0 == null) {
344            chunks0 = new InputChunks(messageId, this.msrp.getMaxContentLength());
345            this.inputChunks.put(messageId, chunks0);
346        }
347        final InputChunks chunks = chunks0;
348
349        // Process request
350        final boolean complete;
351        try {
352            complete = chunks.handleSend(request);
353        } catch (ProtocolException e) {
354            if (!FailureReport.NO.equals(headers.getFailureReport())) {
355                this.outputQueue.add(Session.createMsrpResponse(request,
356                  MsrpConstants.RESPONSE_CODE_BAD_REQUEST, "Protocol error: " + e.getMessage()));
357            }
358            this.close(e);
359            return;
360        }
361
362        // Reply to transaction
363        if (!FailureReport.NO.equals(headers.getFailureReport()) && !FailureReport.PARTIAL.equals(headers.getFailureReport()))
364            this.outputQueue.add(Session.createMsrpResponse(request, MsrpConstants.RESPONSE_CODE_OK, "OK"));
365
366        // Is message aborted?
367        if (chunks.isAborted()) {
368            this.inputChunks.remove(messageId);
369            return;
370        }
371
372        // Is message still incomplete?
373        if (!complete)
374            return;
375
376        // Remove from incoming messages
377        this.inputChunks.remove(messageId);
378
379        // Notify listener of reception of complete message
380        final byte[] content = chunks.getContent();
381        final TreeSet<Header> combinedHeaders = new TreeSet<Header>(Header.SORT_BY_NAME);
382        combinedHeaders.addAll(headers.getMimeHeaders());
383        combinedHeaders.addAll(headers.getExtensionHeaders());
384        this.callbackExecutor.execute(new Runnable() {
385            @Override
386            public void run() {
387                try {
388                    Session.this.listener.sessionReceivedMessage(Session.this, chunks.getFromPath(), messageId,
389                      content, headers.getContentType(), combinedHeaders, chunks.isSuccessReport(),
390                      FailureReport.YES.equals(chunks.getFailureReport()));
391                } catch (ThreadDeath t) {
392                    throw t;
393                } catch (Throwable t) {
394                    Session.this.log.error("error in listener notification", t);
395                }
396            }
397        });
398    }

400    // Handle REPORT request
401    private void handleReport(MsrpRequest request) {
402
403        // Extract info
404        final MsrpHeaders requestHeaders = request.getHeaders();
405        final String messageId = requestHeaders.getMessageId();
406        if (messageId == null) {
407            this.log.warn("rec'd REPORT with no message ID in session from " + this.remoteURI + ", ignoring");
408            return;
409        }
410        final Status status = requestHeaders.getStatus();
411        if (status == null) {
412            this.log.warn("rec'd REPORT with no Status in session from " + this.remoteURI + ", ignoring");
413            return;
414        }
415        final boolean success = status.getNamespace() == 0 && status.getCode() / 100 == 2;
416        final ByteRange byteRange = requestHeaders.getByteRange();
417        if (success && byteRange == null) {
418            this.log.warn("rec'd success REPORT with no ByteRange from " + this.remoteURI + ", ignoring");
419            return;
420        }
421
422        // Find outgoing message
423        final OutputChunks chunks = this.outputChunks.get(messageId);
424        if (chunks == null)
425            return;
426
427        // Notify listener
428        if (success)
429            chunks.notifySuccess(this, this.callbackExecutor, byteRange);
430        else
431            chunks.notifyFailure(this, this.callbackExecutor, status);
432    }

434    // Handle transaction response
435    private void handleResponse(MsrpResponse response) {
436
437        // Find the corresponding transaction and output chunks
438        final OutputTransaction transaction = this.outputTransactions.remove(response.getTransactionId());
439        if (transaction == null)
440            return;
441        final OutputChunks chunks = transaction.getOutputChunks();
442
443        // Success is easy :-)
444        if (response.getCode() < 300)
445            return;
446
447        // Notify failure listener, if any
448        chunks.notifyFailure(this, this.callbackExecutor, response.toStatus());
449
450        // Interrupt output chunks
451        chunks.close();
452
453        // Handle specific codes
454        switch (response.getCode()) {
455        case MsrpConstants.RESPONSE_CODE_SESSION_DOES_NOT_EXIST:
456        case MsrpConstants.RESPONSE_CODE_SESSION_ALREADY_BOUND:
457            this.close(new Exception("rec'd error from remote: " + response.getResultString()));
458            break;
459        default:
460            break;
461        }
462    }

464    /**
465     * Create a {@link MsrpResponse} corresponding to the given request with the given error code and comment.
466     */
467    static MsrpResponse createMsrpResponse(MsrpRequest request, int code, String comment) {
468
469        // Sanity check
470        if (request == null)
471            throw new IllegalArgumentException("null request");
472        new Status(code, comment);
473
474        // Build response
475        final MsrpHeaders requestHeaders = request.getHeaders();
476        final MsrpHeaders responseHeaders = new MsrpHeaders();
477        if (MsrpConstants.METHOD_SEND.equals(request.getMethod()))
478            responseHeaders.getToPath().add(requestHeaders.getFromPath().get(0));
479        else
480            responseHeaders.getToPath().addAll(requestHeaders.getFromPath());
481        responseHeaders.getFromPath().add(requestHeaders.getToPath().get(0));
482        return new MsrpResponse(request.getTransactionId(), code, comment, responseHeaders);
483    }

485    private void enqueueReport(List<MsrpUri> toPath, String messageId, Status status, ByteRange byteRange) {
486        final MsrpHeaders headers = new MsrpHeaders();
487        headers.getToPath().addAll(toPath);
488        headers.getFromPath().add(Session.this.localURI);
489        headers.setMessageId(messageId);
490        headers.setStatus(status);
491        if (byteRange != null)
492            headers.setByteRange(byteRange);
493        this.outputQueue.add(new MsrpRequest(MsrpMessage.randomId(), MsrpConstants.METHOD_REPORT, headers));
494    }

496// Housekeeping
497
498    void performHousekeeping() throws IOException {
499
500        // Create connection if we are active and none exists yet. Note: this can block doing DNS lookups XXX
501        if (this.connection == null && this.activeEndpoint != null)
502            this.connection = this.msrp.createConnection(this.activeEndpoint);
503
504        // If we don't have a connection for longer than the connect timeout, fail
505        if (this.connection == null) {
506            final long age = (System.nanoTime() - this.startTime) / 1000000L;
507            if (age >= this.msrp.getConnectTimeout())
508                throw new IOException("session not bound after " + this.msrp.getConnectTimeout() + "ms");
509        }
510
511        // Add the next chunk of each outstanding message to the output queue in round-robin fashion
512        if (this.outputQueue.isEmpty()) {
513            for (OutputChunks chunks : this.outputChunks.values()) {
514                if (chunks.hasNext()) {
515                    final MsrpMessage request = chunks.next();
516                    this.outputQueue.add(request);
517                    this.outputTransactions.put(request.getTransactionId(),
518                      new OutputTransaction(chunks, request.getTransactionId()));
519                }
520            }
521        }
522
523        // If we have a connection, move enqueued chunks from my output queue to connection's output queue
524        this.flushOutputQueue();
525
526        // Scrub output messages that are complete and either have already been reported on or have timed out waiting
527        for (Iterator<OutputChunks> i = this.outputChunks.values().iterator(); i.hasNext(); ) {
528            final OutputChunks chunks = i.next();
529            if (chunks.hasNext())
530                continue;
531            if (chunks.getReportListener() == null || chunks.getIdleTime() > MAX_MESSAGE_IDLE_TIME_MILLIS)
532                i.remove();
533        }
534
535        // Scrub orphaned input messages
536        for (Iterator<InputChunks> i = this.inputChunks.values().iterator(); i.hasNext(); ) {
537            final InputChunks chunks = i.next();
538            assert !chunks.isComplete() && !chunks.isAborted();
539            if (chunks.getIdleTime() > MAX_MESSAGE_IDLE_TIME_MILLIS) {
540                if (!FailureReport.NO.equals(chunks.getFailureReport())) {
541                    this.enqueueReport(chunks.getFromPath(), chunks.getMessageId(),
542                      new Status(MsrpConstants.RESPONSE_CODE_TIMEOUT, "Missing message chunks never arrived"), null);
543                }
544                i.remove();
545                continue;
546            }
547        }
548
549        // Scrub transactions that have timed out waiting for transaction responses
550        for (Iterator<OutputTransaction> i = this.outputTransactions.values().iterator(); i.hasNext(); ) {
551            final OutputTransaction outputTransaction = i.next();
552            if (outputTransaction.getAge() >= MAX_TRANSACTION_AGE_MILLIS) {
553                outputTransaction.getOutputChunks().notifyFailure(this, this.callbackExecutor,
554                  new Status(MsrpConstants.RESPONSE_CODE_TIMEOUT, "No response rec'd for transaction"));
555                i.remove();
556            }
557        }
558    }
559
560    // If we have a connection, move enqueued chunks from my output queue to connection's output queue
561    private void flushOutputQueue() throws IOException {
562        if (this.connection != null) {
563            for (MsrpMessage message; (message = this.outputQueue.pollFirst()) != null; )
564                this.connection.write(message);
565        }
566    }

570    /**
571     * This object is needed so that when a failure report is requested we can keep track of who to notify.
572     */
573    private class OutputTransaction {
574
575        private final OutputChunks chunks;
576        private final String transactionId;
577        private final long sendTime;
578
579        OutputTransaction(OutputChunks chunks, String transactionId) {
580            this.chunks = chunks;
581            this.transactionId = transactionId;
582            this.sendTime = System.nanoTime();
583        }
584
585        public OutputChunks getOutputChunks() {
586            return this.chunks;
587        }
588
589        public String getTransactionId() {
590            return this.transactionId;
591        }
592
593        public long getAge() {
594            return (System.nanoTime() - this.sendTime) / 1000000L;
595        }
596    }
