006package org.dellroad.msrp.msg;
007
008import java.util.ArrayList;
009import java.util.List;
010import java.util.SortedSet;
011import java.util.TreeSet;
012
013import org.dellroad.msrp.MsrpConstants;
014import org.dellroad.msrp.MsrpUri;
015
016/**
017 * MSRP headers.
018 */
019public class MsrpHeaders {
020
021    private final ArrayList<MsrpUri> toPath = new ArrayList<>();
022    private final ArrayList<MsrpUri> fromPath = new ArrayList<>();
023    private String messageId;
024    private boolean successReport;
025    private FailureReport failureReport = FailureReport.YES;
026    private ByteRange byteRange;
027    private Status status;
028    private String contentType;
029    private final TreeSet<Header> extensionHeaders = new TreeSet<>(Header.SORT_BY_NAME);
030    private final TreeSet<Header> mimeHeaders = new TreeSet<>(Header.SORT_BY_NAME);
031
032    public List<MsrpUri> getToPath() {
033        return this.toPath;
034    }
035
036    public List<MsrpUri> getFromPath() {
037        return this.fromPath;
038    }
039
040    public String getMessageId() {
041        return this.messageId;
042    }
043    public void setMessageId(String messageId) {
044        this.messageId = messageId;
045    }
046
047    public boolean isSuccessReport() {
048        return this.successReport;
049    }
050    public void setSuccessReport(boolean successReport) {
051        this.successReport = successReport;
052    }
053
054    public FailureReport getFailureReport() {
055        return this.failureReport;
056    }
057    public void setFailureReport(FailureReport failureReport) {
058        this.failureReport = failureReport;
059    }
060
061    public ByteRange getByteRange() {
062        return this.byteRange;
063    }
064    public void setByteRange(ByteRange byteRange) {
065        this.byteRange = byteRange;
066    }
067
068    public Status getStatus() {
069        return this.status;
070    }
071    public void setStatus(Status status) {
072        this.status = status;
073    }
074
075    public String getContentType() {
076        return this.contentType;
077    }
078    public void setContentType(String contentType) {
079        this.contentType = contentType;
080    }
081
082    /**
083     * MIME headers, sorted by header name case-insensitively.
084     */
085    public SortedSet<Header> getMimeHeaders() {
086        return this.mimeHeaders;
087    }
088
089    /**
090     * Extension headers, sorted by header name case-insensitively.
091     */
092    public SortedSet<Header> getExtensionHeaders() {
093        return this.extensionHeaders;
094    }
095
096// Object
097
098    @Override
099    public boolean equals(Object obj) {
100        if (obj == this)
101            return true;
102        if (obj == null || obj.getClass() != this.getClass())
103            return false;
104        final MsrpHeaders that = (MsrpHeaders)obj;
105        if (!this.toPath.equals(that.toPath))
106            return false;
107        if (!this.fromPath.equals(that.fromPath))
108            return false;
109        if (!(this.messageId != null ? this.messageId.equals(that.messageId) : that.messageId == null))
110            return false;
111        if (this.successReport != that.successReport)
112            return false;
113        if (this.failureReport != that.failureReport)
114            return false;
115        if (!(this.byteRange != null ? this.byteRange.equals(that.byteRange) : that.byteRange == null))
116            return false;
117        if (!(this.status != null ? this.status.equals(that.status) : that.status == null))
118            return false;
119        if (!(this.contentType != null ? this.contentType.equals(that.contentType) : that.contentType == null))
120            return false;
121        if (!this.extensionHeaders.equals(that.extensionHeaders))
122            return false;
123        if (!this.mimeHeaders.equals(that.mimeHeaders))
124            return false;
125        return true;
126    }
127
128    @Override
129    public int hashCode() {
130        return this.toPath.hashCode()
131          ^ this.fromPath.hashCode()
132          ^ (this.messageId != null ? this.messageId.hashCode() : 0)
133          ^ (this.successReport ? 1 : 0)
134          ^ (this.failureReport != null ?  this.failureReport.hashCode() : 0)
135          ^ (this.byteRange != null ? this.byteRange.hashCode() : 0)
136          ^ (this.status != null ? this.status.hashCode() : 0)
137          ^ (this.contentType != null ? this.contentType.hashCode() : 0)
138          ^ this.extensionHeaders.hashCode()
139          ^ this.mimeHeaders.hashCode();
140    }
141
142    @Override
143    public String toString() {
144        final StringBuilder buf = new StringBuilder();
145        if (!this.toPath.isEmpty()) {
146            buf.append(MsrpConstants.TO_PATH_HEADER).append(": ");
147            for (MsrpUri uri : this.toPath)
148                buf.append(uri);
149            buf.append(Util.CRLF);
150        }
151        if (!this.toPath.isEmpty()) {
152            buf.append(MsrpConstants.FROM_PATH_HEADER).append(": ");
153            for (MsrpUri uri : this.fromPath)
154                buf.append(uri);
155            buf.append(Util.CRLF);
156        }
157        if (this.messageId != null)
158            buf.append(MsrpConstants.MESSAGE_ID_HEADER).append(": ").append(this.messageId).append(Util.CRLF);
159        if (this.successReport)
160            buf.append(MsrpConstants.SUCCESS_REPORT_HEADER).append(": ").append(MsrpConstants.YES_HEADER_VALUE).append(Util.CRLF);
161        if (this.failureReport != null && this.failureReport != FailureReport.YES)
162            buf.append(MsrpConstants.FAILURE_REPORT_HEADER).append(": ").append(this.failureReport).append(Util.CRLF);
163        if (this.byteRange != null)
164            buf.append(MsrpConstants.BYTE_RANGE_HEADER).append(": ").append(this.byteRange).append(Util.CRLF);
165        if (this.status != null)
166            buf.append(MsrpConstants.STATUS_HEADER).append(": ").append(this.status).append(Util.CRLF);
167        for (Header header : this.extensionHeaders)
168            buf.append(header).append(Util.CRLF);
169        for (Header header : this.mimeHeaders)
170            buf.append(header).append(Util.CRLF);
171        if (this.contentType != null)
172            buf.append(MsrpConstants.CONTENT_TYPE_HEADER).append(": ").append(this.contentType).append(Util.CRLF);
173        return buf.toString();
174    }
175}
