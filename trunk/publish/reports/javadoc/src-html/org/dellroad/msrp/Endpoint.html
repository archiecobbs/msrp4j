008package org.dellroad.msrp;
009
010import java.net.InetSocketAddress;
011
012/**
013 * Combination of a host and port.
014 *
015 * <p>
016 * Instances are immutable.
017 * </p>
018 */
019public class Endpoint {
020
021    private final String host;
022    private final int port;
023
024    /**
025     * Constructor.
026     *
027     * @param host remote host
028     * @param port TCP port
029     * @throws IllegalArgumentException if {@code host} is null
030     * @throws IllegalArgumentException if {@code port} is not in the range 1-65535
031     */
032    public Endpoint(String host, int port) {
033        if (host == null)
034            throw new IllegalArgumentException("null host");
035        if (port < 1 || port > 65535)
036            throw new IllegalArgumentException("invalid port " + port);
037        this.host = host;
038        this.port = port;
039    }
040
041    public String getHost() {
042        return this.host;
043    }
044
045    public int getPort() {
046        return this.port;
047    }
048
049    /**
050     * Convert this instance to a {@link InetSocketAddress}. This may result in a DNS lookup.
051     * If the lookup fails, the returned {@link InetSocketAddress} will be <i>unresolved</i>.
052     */
053    public InetSocketAddress toSocketAddress() {
054        return new InetSocketAddress(this.host, this.port);
055    }
056
057    @Override
058    public String toString() {
059        return this.host + ":" + this.port;
060    }
061
062    @Override
063    public boolean equals(Object obj) {
064        if (obj == this)
065            return true;
066        if (obj == null || obj.getClass() != this.getClass())
067            return false;
068        final Endpoint that = (Endpoint)obj;
069        return this.host.equals(that.host) && this.port == that.port;
070    }
071
072    @Override
073    public int hashCode() {
074        return this.host.hashCode() ^ this.port;
075    }
076}
