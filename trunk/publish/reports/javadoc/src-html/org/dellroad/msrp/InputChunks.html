008package org.dellroad.msrp;
009
010import java.util.ArrayList;
011import java.util.Collections;
012import java.util.List;
013import java.util.SortedSet;
014import java.util.TreeSet;
015
016import org.dellroad.msrp.msg.ByteRange;
017import org.dellroad.msrp.msg.FailureReport;
018import org.dellroad.msrp.msg.Header;
019import org.dellroad.msrp.msg.MsrpHeaders;
020import org.dellroad.msrp.msg.MsrpRequest;
021import org.dellroad.msrp.msg.ProtocolException;
022
023/**
024 * Represents one incoming MSRP message reassembled from multiple {@code SEND} {@link MsrpRequest} chunks.
025 */
026public class InputChunks {
027
028    private final String messageId;
029    private final long maxContentLength;
030    private final TreeSet<Header> headers = new TreeSet<Header>(Header.SORT_BY_NAME);
031    private final ArrayList<long[]> holes = new ArrayList<>();
032
033    private List<MsrpUri> fromPath;
034    private boolean successReport;
035    private FailureReport failureReport = FailureReport.YES;
036    private String contentType;
037    private boolean complete;
038    private boolean aborted;
039    private long contentLength = -1;
040    private long timestamp;
041    private byte[] buf;
042
043    /**
044     * Constructor.
045     *
046     * @param messageId incoming message ID
047     * @param maxContentLength maximum allowed message content length
048     * @throws IllegalArgumentException if {@code messageId} is null
049     */
050    public InputChunks(String messageId, long maxContentLength) {
051        if (messageId == null)
052            throw new IllegalArgumentException("null messageId");
053        this.messageId = messageId;
054        this.maxContentLength = maxContentLength;
055        this.timestamp = System.nanoTime();
056    }
057
058    /**
059     * Handle an incoming SEND request for this message. Also resets this instance's idle timer.
060     *
061     * @param request incoming SEND request associated with this message
062     * @return true if this message is now complete (or aborted); false if more chunks are still required
063     * @throws IllegalArgumentException if {@code request} is null
064     * @throws IllegalArgumentException if {@code request} is not a {@code SEND} request
065     * @throws IllegalArgumentException if {@code request} has the wrong message ID
066     * @throws ProtocolException if a protocol violation is detected
067     */
068    public synchronized boolean handleSend(MsrpRequest request) throws ProtocolException {
069
070        // Sanity check
071        if (request == null)
072            throw new IllegalArgumentException("null request");
073        if (!MsrpConstants.METHOD_SEND.equals(request.getMethod()))
074            throw new IllegalArgumentException("request method " + request.getMethod() + " != " + MsrpConstants.METHOD_SEND);
075        final MsrpHeaders requestHeaders = request.getHeaders();
076        if (!this.messageId.equals(requestHeaders.getMessageId()))
077            throw new IllegalArgumentException("request message ID " + requestHeaders.getMessageId() + " != " + this.messageId);
078
079        // Update timestamp
080        this.timestamp = System.nanoTime();
081
082        // Get From-Path and success/failure report info
083        this.fromPath = requestHeaders.getFromPath();
084        this.successReport |= requestHeaders.isSuccessReport();
085        if (requestHeaders.getFailureReport() != null)
086            this.failureReport = requestHeaders.getFailureReport();
087
088        // Get extension header info
089        this.headers.addAll(requestHeaders.getExtensionHeaders());
090
091        // Handle aborted message
092        if (request.isAborted()) {
093            this.aborted = true;
094            return true;
095        }
096
097        // Get body; if none, check sanity
098        final byte[] body = request.getBody();
099        if (body == null) {
100            if (this.buf != null)
101                throw new ProtocolException("continuation request must have a body");
102            final ByteRange byteRange = requestHeaders.getByteRange();
103            if (byteRange != null && !byteRange.equals(ByteRange.EMPTY))
104                throw new ProtocolException("invalid ByteRange " + byteRange + " for message having no body");
105            this.complete = true;
106            return true;
107        }
108
109        // Get MIME header info
110        this.contentType = requestHeaders.getContentType();
111        this.headers.addAll(requestHeaders.getMimeHeaders());
112
113        // Get/infer byte range and validate
114        ByteRange byteRange = requestHeaders.getByteRange();
115        if (byteRange == null)
116            byteRange = new ByteRange(body.length);
117        final long expectedEnd = byteRange.getStart() + body.length - 1;
118        if (byteRange.getEnd() == -1)
119            byteRange = new ByteRange(byteRange.getStart(), expectedEnd, byteRange.getTotal());
120        else if (byteRange.getEnd() != expectedEnd) {
121            throw new ProtocolException("rec'd inconsistent ByteRange " + byteRange + " with end byte "
122              + byteRange.getEnd() + " != " + expectedEnd + " expected based on body size");
123        }
124
125        // Determine the range of content bytes we just received
126        final long offset = byteRange.getStart() - 1;
127        final long limit = offset + body.length;
128
129        // Infer/validate content length
130        if (this.contentLength == -1) {
131            if (byteRange.getTotal() != -1)
132                this.contentLength = byteRange.getTotal();
133        } else if (byteRange.getTotal() != -1 && byteRange.getTotal() != this.contentLength) {
134            throw new ProtocolException("rec'd inconsistent ByteRange " + byteRange
135              + " with total " + byteRange.getTotal() + " != previously rec'd total " + this.contentLength);
136        }
137
138        // Verify final chunk is not marked incomplete
139        if (this.contentLength != -1 && byteRange.getEnd() == this.contentLength && !request.isComplete())
140            throw new ProtocolException("last chunk in " + byteRange + " message has unexpected incomplete flag");
141
142        // Check content length not too big
143        final long minimumLength = Math.max(limit, this.contentLength);
144        if (minimumLength > this.maxContentLength || minimumLength > Integer.MAX_VALUE)     // TODO: allow long length with stream
145            throw new ProtocolException("content is too large (" + minimumLength + " > " + this.maxContentLength + " bytes)");
146
147        // Create content buffer if it doesn't already exist
148        if (this.buf == null)
149            this.buf = new byte[0];
150
151        // Expand content buffer as needed and add a corresponding hole TODO: allow streaming to a file instead of into memory
152        if (this.buf.length < minimumLength) {
153            this.holes.add(new long[] { this.buf.length, minimumLength });
154            final byte[] newBuf = new byte[(int)minimumLength];
155            System.arraycopy(this.buf, 0, newBuf, 0, this.buf.length);
156            this.buf = newBuf;
157        }
158
159        // Merge new data into content buffer and update holes
160        System.arraycopy(body, 0, this.buf, (int)offset, body.length);
161        for (int i = 0; i < this.holes.size(); i++) {
162            final long[] hole = this.holes.get(i);
163            assert hole[0] < hole[1];
164            if (hole[1] <= offset || hole[0] >= limit)      // entirely before or after body (i.e., not a factor)
165                continue;
166            if (hole[0] >= offset && hole[1] <= limit)      // entirely contained within body
167                this.holes.remove(i--);
168            else if (hole[1] <= limit)                      // straddles body's left border
169                hole[1] = offset;
170            else if (hole[0] >= offset)                     // straddles body's right border
171                hole[0] = limit;
172            else {                                          // entirely contains body => split it in two
173                this.holes.set(i, new long[] { hole[0], offset });
174                this.holes.add(++i, new long[] { limit, hole[1] });
175            }
176        }
177
178        // Are we complete now?
179        this.complete |= request.isComplete() && this.holes.isEmpty();
180
181        // Done
182        return this.complete;
183    }
184
185    /**
186     * Get idle time of this instance.
187     *
188     * @return time in milliseconds since (construction or) the most recent invocation of {@link #handleSend handleSend()}
189     */
190    public synchronized long getIdleTime() {
191        return (System.nanoTime() - this.timestamp) / 1000000L;
192    }
193
194    /**
195     * Get message ID.
196     */
197    public synchronized String getMessageId() {
198        return this.messageId;
199    }
200
201    /**
202     * Get From path.
203     */
204    public synchronized List<MsrpUri> getFromPath() {
205        return this.fromPath;
206    }
207
208    /**
209     * Get message content.
210     *
211     * @return message content, or null if this message does not contain any content
212     */
213    public synchronized byte[] getContent() {
214        return this.buf;
215    }
216
217    /**
218     * Get message content type.
219     *
220     * @return message content type, or null if this message does not contain any content
221     */
222    public synchronized String getContentType() {
223        return this.contentType;
224    }
225
226    /**
227     * Get other headers, sorted by header name case-insensitively.
228     */
229    public synchronized SortedSet<Header> getHeaders() {
230        return Collections.<Header>unmodifiableSortedSet(this.headers);
231    }
232
233    /**
234     * Determine whether this instance is complete.
235     */
236    public synchronized boolean isComplete() {
237        return this.complete;
238    }
239
240    /**
241     * Determine whether this instance was aborted.
242     */
243    public synchronized boolean isAborted() {
244        return this.aborted;
245    }
246
247    /**
248     * Determine whether this message requires a success report.
249     */
250    public synchronized boolean isSuccessReport() {
251        return this.successReport;
252    }
253
254    /**
255     * Determine what type of failure reporting this message requires.
256     */
257    public synchronized FailureReport getFailureReport() {
258        return this.failureReport;
259    }
260}
