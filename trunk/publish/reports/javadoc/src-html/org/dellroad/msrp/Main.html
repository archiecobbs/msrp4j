044/**
045 * Command line utility.
046 *
047 * <p>
048 * This class depends on the <a href="https://github.com/jline/jline2">JLine</a> and
049 * <a href="http://dellroad-stuff.googlecode.com/">dellroad-stuff</a> Java libraries.

052public class Main extends MainClass {
053
054    private static final Charset UTF8 = Charset.forName("UTF-8");
055
056    private final Logger log = LoggerFactory.getLogger(this.getClass());
057    private final SessionListener sessionListener = new MainSessionListener();
058    private final MainReportListener reportListener = new MainReportListener();
059    private final ScheduledExecutorService executor = Executors.newSingleThreadScheduledExecutor();
060    private final Msrp msrp = new Msrp();
061
062    private int port = MsrpConstants.DEFAULT_PORT;
063    private ConsoleReader console;
064    private PrintWriter writer;
065    private FileHistory history;
066    private CursorBuffer cursorBuffer;
067    private boolean verbose;
068
069    private MsrpUri currentSession;

071    @Override
072    public int run(String[] args) throws Exception {
073
074        // Parse command line
075        final ArrayDeque<String> params = new ArrayDeque<String>(Arrays.asList(args));
076        while (!params.isEmpty() && params.peekFirst().startsWith("-")) {
077            final String option = params.removeFirst();
078            if (option.equals("-h") || option.equals("--help")) {
079                this.usageMessage();
080                return 0;
081            } else if (option.equals("-v") || option.equals("--verbose"))
082                this.verbose = true;
083            else if (option.equals("--port"))
084                this.port = this.parseIntParam(params, "port");
085            else if (option.equals("--max-sessions"))
086                this.msrp.setMaxSessions(this.parseIntParam(params, "max-sessions"));
087            else if (option.equals("--max-content-length"))
088                this.msrp.setMaxContentLength(this.parseIntParam(params, "max-content-length"));
089            else if (option.equals("--idle-timeout"))
090                this.msrp.setMaxIdleTime(this.parseIntParam(params, "idle-timeout") * 1000L);
091            else if (option.equals("--connect-timeout"))
092                this.msrp.setConnectTimeout(this.parseIntParam(params, "connect-timeout") * 1000L);
093            else if (option.equals("--"))
094                break;
095            else {
096                System.err.println(this.getName() + ": unknown option `" + option + "'");
097                this.usageError();
098            }
099        }
100        switch (params.size()) {
101        case 0:
102            break;
103        default:
104            this.usageError();
105            return 1;
106        }
107
108        // Set up console
109        this.console = new ConsoleReader(new FileInputStream(FileDescriptor.in), System.out);
110        this.console.setBellEnabled(true);
111        this.console.setHistoryEnabled(true);
112        this.console.setHandleUserInterrupt(true);
113        this.writer = new PrintWriter(console.getOutput(), true);
114        try {
115            this.history = new FileHistory(new File(new File(System.getProperty("user.home")), ".msrp_history"));
116        } catch (IOException e) {
117            // ignore
118        }
119        this.console.setHistory(this.history);
120
121        // Start up MSRP stack
122        this.msrp.setListenAddress(new InetSocketAddress(this.port));
123        this.msrp.start();
124
125        // Main command loop
126        this.writer.println("Welcome to MSRP. Type `help' for help.");
127        this.writer.println("Listening on address " + this.msrp.getListenAddress());
128        final StringBuilder lineBuffer = new StringBuilder();
129        try {
130            for (boolean done = false; !done; ) {
131
132                // Read command line
133                String line;
134                try {
135                    line = this.console.readLine(lineBuffer.length() == 0 ? "MSRP> " : "   -> ");
136                } catch (UserInterruptException e) {
137                    this.writer.print("^C");
138                    line = null;
139                }
140                if (line == null) {
141                    this.writer.println();
142                    break;
143                }
144
145                // Detect backslash continuations
146                boolean continuation = false;
147                if (line.length() > 0 && line.charAt(line.length() - 1) == '\\') {
148                    line = line.substring(0, line.length() - 1) + "\n";
149                    continuation = true;
150                }
151
152                // Append line to buffer
153                lineBuffer.append(line);
154
155                // Handle backslash continuations
156                if (continuation)
157                    continue;
158                final ParseContext ctx = new ParseContext(lineBuffer.toString());
159                lineBuffer.setLength(0);
160
161                // Skip initial whitespace
162                ctx.skipWhitespace();
163
164                // Ignore blank input
165                if (ctx.getInput().length() == 0)
166                    continue;
167
168                // Get command and handle
169                final String command = ctx.matchPrefix("[^\\s]+").group();
170                ctx.skipWhitespace();
171                try {
172                    switch (command) {
173                    case "connect":
174                    case "accept":
175                    {
176                        final MsrpUri localURI = new MsrpUri(ctx.matchPrefix("[^\\s]+").group());
177                        ctx.skipWhitespace();
178                        final MsrpUri remoteURI = new MsrpUri(ctx.matchPrefix("[^\\s]+").group());
179                        this.open(localURI, remoteURI, command.equals("connect"));
180                        break;
181                    }
182                    case "close":
183                        this.close();
184                        break;
185                    case "list":
186                        this.list();
187                        break;
188                    case "select":
189                        this.currentSession = new MsrpUri(ctx.matchPrefix("[^\\s]+").group());
190                        this.writer.println("* New current session is " + this.currentSession);
191                        break;
192                    case "text":
193                        this.send(new ByteArrayInputStream(ctx.getInput().getBytes(UTF8)), "text/plain");
194                        break;
195                    case "send":
196                    {
197                        if (ctx.getInput().length() == 0) {
198                            this.send(null, null);
199                            break;
200                        }
201                        final File file = new File(ctx.matchPrefix("[^\\s]+").group());
202                        ctx.skipWhitespace();
203                        final String contentType = ctx.matchPrefix("[^\\s]+").group();
204                        this.send(new FileInputStream(file), contentType);
205                        break;
206                    }
207                    case "success":
208                    {
209                        final String messageId = ctx.matchPrefix("[^\\s]+").group();
210                        ctx.skipWhitespace();
211                        final ByteRange byteRange = ctx.getInput().length() != 0 ?
212                          ByteRange.fromString(ctx.getInput()) : ByteRange.ALL;
213                        this.success(messageId, byteRange);
214                        break;
215                    }
216                    case "failure":
217                    {
218                        final String messageId = ctx.matchPrefix("[^\\s]+").group();
219                        ctx.skipWhitespace();
220                        this.failure(messageId, Status.fromString(ctx.getInput()));
221                        break;
222                    }
223                    case "quit":
224                        this.writer.println("* Bye");
225                        done = true;
226                        break;
227                    case "help":
228                        this.writer.println("Available commands:");
229                        this.writer.println("  connect localURI remoteURI");
230                        this.writer.println("      Create an active session using the given URIs");
231                        this.writer.println("  accept localURI remoteURI");
232                        this.writer.println("      Create a passive session using the given URIs");
233                        this.writer.println("  close");
234                        this.writer.println("      Shutdown selected session");
235                        this.writer.println("  list");
236                        this.writer.println("      List all known sessions");
237                        this.writer.println("  select localURI");
238                        this.writer.println("      Select current session");
239                        this.writer.println("  text ...");
240                        this.writer.println("      Send a text message");
241                        this.writer.println("  send");
242                        this.writer.println("      Send a message with no content");
243                        this.writer.println("  send name content-type");
244                        this.writer.println("      Send arbitrary file");
245                        this.writer.println("  success messageId [byte-range]");
246                        this.writer.println("      Send success report");
247                        this.writer.println("  failure messageId nscode code [comment]");
248                        this.writer.println("      Send failure report");
249                        this.writer.println("  quit");
250                        this.writer.println("      Close all sessions and quit");
251                        this.writer.println("  help");
252                        this.writer.println("      Show commands");
253                        break;
254                    default:
255                        throw new Exception("unknown command `" + command + "'; type `help' for help.");
256                    }
257                } catch (Exception e) {
258                    final String msg = e.getMessage() != null ? e.getMessage() : "" + e;
259                    this.writer.println("Error: " + msg);
260                    if (this.verbose)
261                        e.printStackTrace(this.writer);
262                }
263
264                // Flush output
265                this.writer.flush();
266            }
267        } finally {
268            if (this.history != null)
269                this.history.flush();
270            this.writer.flush();
271            this.console.flush();
272            this.console.shutdown();
273        }
274
275        // Stop MSRP
276        this.msrp.stop();
277
278        // Done
279        return 0;
280    }
281
282    private void stashLine() {
283        this.cursorBuffer = this.console.getCursorBuffer().copy();
284        try {
285            this.console.getOutput().write("\u001b[1G\u001b[K");
286            this.console.flush();
287        } catch (IOException e) {
288            // ignore
289        }
290    }
291
292    private void unstashLine() {
293        try {
294            this.console.resetPromptLine(this.console.getPrompt(), this.cursorBuffer.toString(), this.cursorBuffer.cursor);
295        } catch (IOException e) {
296            // ignore
297        }
298    }

308        System.err.println("Options:");
309        System.err.println("  --port port           Port for incoming connections (default " + MsrpConstants.DEFAULT_PORT + ")");

311        System.err.println("  --max-sessions        Set maximum allowed number of sessions");
312        System.err.println("  --max-content-length  Set maximum allowed message content length");
313        System.err.println("  --idle-timeout        Set maximum allowed time for idle connections (in seconds)");
314        System.err.println("  --connect-timeout     Set connection timeout for outbound connections (in seconds)");
315    }

321    private int parseIntParam(ArrayDeque<String> params, String name) {
322        if (params.isEmpty()) {
323            this.usageError();
324            return 0;
325        }
326        final String string = params.removeFirst();
327        try {
328            return Integer.parseInt(string, 10);
329        } catch (Exception e) {
330            System.err.println(this.getName() + ": invalid " + name + " `" + string + "'");
331            this.usageError();
332            return 0;
333        }
334    }

338    private void open(MsrpUri localURI, MsrpUri remoteURI, boolean active) throws Exception {
339        final Session session = this.msrp.createSession(localURI, remoteURI, null, this.sessionListener, this.executor, active);
340        Main.this.writer.println("* Created session: local=" + session.getLocalUri() + " remote=" + session.getRemoteUri());
341        Main.this.writer.println("* New current session is " + session.getLocalUri());
342        this.currentSession = localURI;
343    }
344
345    private void send(InputStream input, String contentType) throws Exception {
346        final Session session = this.findCurrentSession();
347        if (input == null) {
348            session.send(null, this.reportListener);
349            return;
350        }
351        final String messageId = session.send(input, -1, contentType, null, this.reportListener);
352        Main.this.writer.println("* Sent message with message ID " + messageId);
353    }
354
355    private Session findCurrentSession() {
356        final Session session = this.msrp.getSessions().get(this.currentSession);
357        if (session == null)
358            throw new RuntimeException("no session found corresponding to local URI " + this.currentSession);
359        return session;
360    }
361
362    private void close() {
363        final Session session = this.findCurrentSession();
364        this.writer.println("* Closing session " + session.getLocalUri());
365        session.close(null);
366    }
367
368    private void list() {
369        final SortedMap<MsrpUri, Session> sessionMap = this.msrp.getSessions();
370        this.writer.println("* " + sessionMap.size() + " active sessions:");
371        for (Session session : sessionMap.values())
372            this.writer.println("* localURI=" + session.getLocalUri() + " remoteURI=" + session.getRemoteUri());
373    }
374
375    private void success(String messageId, ByteRange byteRange) {
376        final Session session = this.findCurrentSession();
377        this.writer.println("* Sending success report to " + session.getRemoteUri());
378        session.sendSuccessReport(Collections.singletonList(session.getRemoteUri()), messageId, byteRange, null);
379    }
380
381    private void failure(String messageId, Status status) {
382        final Session session = this.findCurrentSession();
383        this.writer.println("* Sending failure report to " + session.getRemoteUri());
384        session.sendFailureReport(Collections.singletonList(session.getRemoteUri()), messageId, status);
385    }

389    private class MainSessionListener implements SessionListener {
390
391        @Override
392        public void sessionClosed(Session session, Exception cause) {
393            Main.this.stashLine();
394            Main.this.writer.println("* Session " + session.getLocalUri() + " closed: " + cause);
395            Main.this.unstashLine();
396        }
397
398        @Override
399        public void sessionReceivedMessage(Session session, List<MsrpUri> fromPath, String messageId, byte[] content,
400          String contentType, SortedSet<Header> headers, boolean successReport, boolean failureReport) {
401            Main.this.stashLine();
402            Main.this.writer.println("* Rec'd message from " + session.getRemoteUri() + ":");
403            Main.this.writer.println("  Message-ID: " + messageId);
404            Main.this.writer.println("  Success-Report: " + (successReport ? "yes" : "no"));
405            Main.this.writer.println("  Failure-Report: " + (failureReport ? "yes" : "no"));
406            for (Header header : headers)
407                Main.this.writer.println("  " + header);
408            if (content == null)
409                Main.this.writer.println("  [ Message contains no content ]");
410            else {
411                Main.this.writer.println("  Content-Type: " + contentType);
412                final MessageDigest sha1;
413                try {
414                    sha1 = MessageDigest.getInstance("SHA");
415                } catch (NoSuchAlgorithmException e) {
416                    throw new RuntimeException(e);
417                }
418                Main.this.writer.println("  Actual Length: " + content.length + " bytes");
419                Main.this.writer.println("  Content SHA1: " + ByteArrayEncoder.encode(sha1.digest(content)));
420                if (contentType.equals("text/plain")) {
421                    Main.this.writer.println();
422                    Main.this.writer.println(new String(content, UTF8));
423                    Main.this.writer.println();
424                }
425            }
426            Main.this.unstashLine();
427        }
428    }

432    private class MainReportListener implements SuccessListener, FailureListener {
433
434        @Override
435        public void reportFailure(Session session, String messageId, Status status) {
436            Main.this.stashLine();
437            Main.this.writer.println("* Rec'd failure from " + session.getRemoteUri() + ":");
438            Main.this.writer.println("  Message-ID: " + messageId);
439            Main.this.writer.println("  Status: " + status);
440            Main.this.unstashLine();
441        }
442
443        @Override
444        public void reportSuccess(Session session, String messageId, ByteRange byteRange) {
445            Main.this.stashLine();
446            Main.this.writer.println("* Rec'd success from " + session.getRemoteUri() + ":");
447            Main.this.writer.println("  Message-ID: " + messageId);
448            Main.this.writer.println("  ByteRange: " + byteRange);
449            Main.this.unstashLine();
450        }
451    }
