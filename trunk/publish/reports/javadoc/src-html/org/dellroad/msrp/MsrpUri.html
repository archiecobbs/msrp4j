008package org.dellroad.msrp;
009
010import java.net.URI;
011import java.net.URISyntaxException;
012import java.util.ArrayList;
013import java.util.Collections;
014import java.util.LinkedHashSet;
015import java.util.List;
016import java.util.Set;
017
018/**
019 * Represents an MRSP URI. This class only supports URIs that have session ID's.
020 */
021public class MsrpUri {
022
023    private final URI uri;
024    private final String sessionId;
025    private final String transport;
026    private final Set<String> parameters;
027
028    /**
029     * Constrcuctor.
030     *
031     * @throws URISyntaxException if {@code string} is not a valid MSRP URI
032     */
033    public MsrpUri(String string) throws URISyntaxException {
034
035        // Parse URI
036        this.uri = new URI(string);
037
038        // Basic structural checking
039        final String scheme = this.uri.getScheme();
040        if (scheme == null)
041            throw new URISyntaxException(string, "invalid MSRP URI: no scheme");
042        if (!MsrpConstants.MSRP_SCHEME.equalsIgnoreCase(scheme) && !MsrpConstants.MSRPS_SCHEME.equalsIgnoreCase(scheme))
043            throw new URISyntaxException(string, "invalid MSRP URI: unknown scheme `" + scheme + "'");
044        if (this.uri.getAuthority() == null)
045            throw new URISyntaxException(string, "invalid MSRP URI: missing authority");
046        if (this.uri.getQuery() != null)
047            throw new URISyntaxException(string, "invalid MSRP URI: query string is not allowed");
048        if (this.uri.getFragment() != null)
049            throw new URISyntaxException(string, "invalid MSRP URI: fragment is not allowed");
050        final String path = this.uri.getPath();
051        if (path == null || path.length() == 0 || path.charAt(0) != '/')
052            throw new URISyntaxException(string, "invalid MSRP URI: no path, or path does not start with `/'");
053
054        // Parse session ID, transport, and parameters out from path
055        final List<String> plist = new ArrayList<>(3);
056        for (int pos = 1; pos <= path.length(); ) {
057            int end = path.indexOf(';', pos);
058            if (end == -1)
059                end = path.length();
060            plist.add(path.substring(pos, end));
061            pos = end + 1;
062        }
063        switch (plist.size()) {
064        case 0:
065            throw new URISyntaxException(string, "invalid MSRP URI: no session ID specified");
066        case 1:
067            throw new URISyntaxException(string, "invalid MSRP URI: no transport specified");
068        default:
069            break;
070        }
071        this.sessionId = plist.get(0);
072        this.transport = plist.get(1);
073        this.parameters = Collections.unmodifiableSet(new LinkedHashSet<String>(plist.subList(2, plist.size())));
074
075        // Verify transport is "tcp"
076        if (!this.transport.equals(MsrpConstants.TRANSPORT_TCP))
077            throw new URISyntaxException(string, "invalid MSRP URI: unknown transport `" + this.transport + "'");
078    }
079
080    /**
081     * Get this instance as an {@link URI}.
082     */
083    public URI getUri() {
084        return this.uri;
085    }
086
087    /**
088     * Determine whether this URI uses TLS.
089     */
090    public boolean isSecure() {
091        return MsrpConstants.MSRPS_SCHEME.equalsIgnoreCase(this.uri.getScheme());
092    }
093
094    /**
095     * Get the MSRP session ID specified in this URI.
096     */
097    public String getSessionId() {
098        return this.sessionId;
099    }
100
101    /**
102     * Get the MSRP transport specified in this URI. Currently will always be {@link MsrpConstants#TRANSPORT_TCP}.
103     */
104    public String getTransport() {
105        return this.transport;
106    }
107
108    /**
109     * Get the URI-parameters, if any. The returned {@link Set} will iterate the parameters in their original order.
110     *
111     * @return unmodifiable set of URI-parameters, possibly empty
112     */
113    public Set<String> getParameters() {
114        return this.parameters;
115    }
116
117    /**
118     * Create an {@link Endpoint} corresponding to this instance.
119     */
120    public Endpoint toEndpoint() {
121        int port = this.uri.getPort();
122        if (port == -1)
123            port = MsrpConstants.DEFAULT_PORT;
124        return new Endpoint(this.uri.getHost(), port);
125    }
126
127// Object
128
129    @Override
130    public String toString() {
131        return this.uri.toString();
132    }
133
134    @Override
135    public boolean equals(Object obj) {
136        if (obj == this)
137            return true;
138        if (obj == null || obj.getClass() != this.getClass())
139            return false;
140        final MsrpUri that = (MsrpUri)obj;
141        return this.uri.equals(that.uri);
142    }
143
144    @Override
145    public int hashCode() {
146        return this.uri.hashCode();
147    }
148}
