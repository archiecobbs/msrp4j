008package org.dellroad.msrp;
009
010import java.net.URI;
011import java.util.Comparator;
012
013/**
014 * Compares MRSP URI's for equality according to RFC 4579, Section 6.1.
015 */
016public final class MsrpUriComparator implements Comparator<MsrpUri> {
017
018    public static final MsrpUriComparator INSTANCE = new MsrpUriComparator();
019
020    private MsrpUriComparator() {
021    }
022
023    @Override
024    public int compare(MsrpUri msrp1, MsrpUri msrp2) {
025        final URI uri1 = msrp1.getUri();
026        final URI uri2 = msrp2.getUri();
027        int diff = uri1.getScheme().toLowerCase().compareTo(uri2.getScheme().toLowerCase());
028        if (diff != 0)
029            return diff;
030        diff = uri1.getHost().toLowerCase().compareTo(uri2.getHost().toLowerCase());
031        if (diff != 0)
032            return diff;
033        diff = Integer.compare(uri1.getPort(), uri2.getPort());
034        if (diff != 0)
035            return diff;
036        diff = msrp1.getSessionId().compareTo(msrp2.getSessionId());
037        if (diff != 0)
038            return diff;
039        diff = msrp1.getTransport().compareTo(msrp2.getTransport());
040        if (diff != 0)
041            return diff;
042        return 0;
043    }
044}
