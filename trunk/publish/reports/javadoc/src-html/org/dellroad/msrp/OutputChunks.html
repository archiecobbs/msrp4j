008package org.dellroad.msrp;
009
010import java.io.ByteArrayOutputStream;
011import java.io.Closeable;
012import java.io.IOException;
013import java.io.InputStream;
014import java.util.Iterator;
015import java.util.NoSuchElementException;
016import java.util.concurrent.Executor;
017
018import org.dellroad.msrp.msg.ByteRange;
019import org.dellroad.msrp.msg.FailureReport;
020import org.dellroad.msrp.msg.Header;
021import org.dellroad.msrp.msg.MsrpHeaders;
022import org.dellroad.msrp.msg.MsrpMessage;
023import org.dellroad.msrp.msg.MsrpRequest;
024import org.dellroad.msrp.msg.Status;
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028/**
029 * Represents one outgoing MSRP message broken into multiple {@code SEND} {@link MsrpRequest} chunks.
030 */
031public class OutputChunks implements Closeable, Iterator<MsrpRequest> {
032
033    private static final int MAX_REQUEST_BODY_SIZE = 2048;
034
035    private final Logger log = LoggerFactory.getLogger(this.getClass());
036    private final MsrpUri localURI;
037    private final MsrpUri remoteURI;
038    private final String messageId = MsrpMessage.randomId();
039    private final Iterable<? extends Header> headers;
040    private final InputStream input;
041    private final String contentType;
042    private final long size;
043
044    private ReportListener reportListener;
045    private long offset;
046    private boolean complete;
047    private boolean aborted;
048    private boolean closed;
049    private long timestamp;
050
051    /**
052     * Constructor for a message with no body.
053     */
054    public OutputChunks(MsrpUri localURI, MsrpUri remoteURI, Iterable<? extends Header> headers, ReportListener reportListener) {
055        this(localURI, remoteURI, null, -1, null, headers, reportListener);
056    }
057
058    /**
059     * Constructor for a message with a body.
060     *
061     * @param localURI local MSRP URI
062     * @param remoteURI remote MSRP URI
063     * @param input message content input, or null for no content
064     * @param size size of input, or -1 if unknown; if not -1 and {@code input} is longer than {@code size}, it will be truncated
065     * @param contentType content type
066     * @param headers other headers (MIME and/or extension), or null for none
067     * @param reportListener success and/or failure listener, or null for none
068     * @throws IllegalArgumentException if {@code localURI} or {@code remoteURI} is null
069     * @throws IllegalArgumentException if {@code input} is null and {@code size != -1} or {@code contentType} is not null
070     * @throws IllegalArgumentException if {@code input} is not null and {@code contentType} is null
071     */
072    public OutputChunks(MsrpUri localURI, MsrpUri remoteURI, InputStream input, long size,
073      String contentType, Iterable<? extends Header> headers, ReportListener reportListener) {
074        if (localURI == null)
075            throw new IllegalArgumentException("null localURI");
076        if (remoteURI == null)
077            throw new IllegalArgumentException("null remoteURI");
078        if (input == null) {
079            if (size != -1)
080                throw new IllegalArgumentException("null input requires size = -1");
081            if (contentType != null)
082                throw new IllegalArgumentException("null input requires null contentType");
083        } else {
084            if (size < -1)
085                throw new IllegalArgumentException("invalid size " + size);
086            if (contentType == null)
087                throw new IllegalArgumentException("null contentType");
088        }
089        this.localURI = localURI;
090        this.remoteURI = remoteURI;
091        this.input = input;
092        this.size = size;
093        this.contentType = contentType;
094        this.reportListener = reportListener;
095        this.headers = headers;
096        this.timestamp = System.nanoTime();
097    }
098
099    /**
100     * Get the unique message ID for this message.
101     */
102    public String getMessageId() {
103        return this.messageId;
104    }
105
106    /**
107     * Get the associated {@link ReportListener}, if any.
108     */
109    public synchronized ReportListener getReportListener() {
110        return this.reportListener;
111    }
112
113    /**
114     * Get the size of the message body, if known.
115     *
116     * @return message body size, or -1 if size was not specified and message has not been fully sent yet
117     */
118    public synchronized long getSize() {
119        return this.size == -1 && this.complete ? this.offset : this.size;
120    }
121
122    /**
123     * Determine whether this instance has been aborted.
124     */
125    public synchronized boolean isAborted() {
126        return this.aborted;
127    }
128
129    public synchronized long getIdleTime() {
130        return (System.nanoTime() - this.timestamp) / 1000000L;
131    }
132
133    /**
134     * Notify about success, if appropriate.
135     *
136     * @param session session on which the message was transmitted
137     * @param executor executor used to issue notification
138     * @param byteRange range of bytes successfully received
139     */
140    public synchronized void notifySuccess(final Session session, Executor executor, final ByteRange byteRange) {
141
142        // Sanity check
143        if (executor == null)
144            throw new IllegalArgumentException("null executor");
145        if (byteRange == null)
146            throw new IllegalArgumentException("null byteRange");
147
148        // Is success notification desired?
149        if (!(this.reportListener instanceof SuccessListener))
150            return;
151        final SuccessListener successListener = (SuccessListener)this.reportListener;
152
153        // Notify
154        executor.execute(new Runnable() {
155            @Override
156            public void run() {
157                try {
158                    successListener.reportSuccess(session, OutputChunks.this.messageId, byteRange);
159                } catch (ThreadDeath t) {
160                    throw t;
161                } catch (Throwable t) {
162                    OutputChunks.this.log.error("error in listener notification", t);
163                }
164            }
165        });
166    }
167
168    /**
169     * Notify about failure, if appropriate. This method is idempotent.
170     *
171     * @param session session on which the message was transmitted
172     * @param executor executor used to issue notification
173     * @param status failure status
174     */
175    public synchronized void notifyFailure(final Session session, Executor executor, final Status status) {
176
177        // Sanity check
178        if (executor == null)
179            throw new IllegalArgumentException("null executor");
180        if (status == null)
181            throw new IllegalArgumentException("null status");
182
183        // Is failure notification desired?
184        if (!(this.reportListener instanceof FailureListener))
185            return;
186        final FailureListener failureListener = (FailureListener)this.reportListener;
187
188        // Only notify once
189        this.reportListener = null;
190
191        // Notify
192        executor.execute(new Runnable() {
193            @Override
194            public void run() {
195                try {
196                    failureListener.reportFailure(session, OutputChunks.this.messageId, status);
197                } catch (ThreadDeath t) {
198                    throw t;
199                } catch (Throwable t) {
200                    OutputChunks.this.log.error("error in listener notification", t);
201                }
202            }
203        });
204    }
205
206// Iterator
207
208    @Override
209    public synchronized boolean hasNext() {
210        return !this.complete;
211    }
212
213    @Override
214    public synchronized MsrpRequest next() {
215
216        // Already done?
217        if (this.complete)
218            throw new NoSuchElementException();
219
220        // Update timestamp
221        this.timestamp = System.nanoTime();
222
223        // Generate new transaction ID
224        final String transactionId = MsrpMessage.randomId();
225
226        // Prepare MSRP request (without body yet)
227        final MsrpHeaders msrpHeaders = new MsrpHeaders();
228        msrpHeaders.getFromPath().add(this.localURI);
229        msrpHeaders.getToPath().add(this.remoteURI);
230        msrpHeaders.setMessageId(this.messageId);
231        msrpHeaders.setContentType(this.contentType);
232        msrpHeaders.setSuccessReport(false);
233        msrpHeaders.setSuccessReport(this.reportListener instanceof SuccessListener);
234        if (this.reportListener instanceof FailureListener)
235            msrpHeaders.setFailureReport(FailureReport.YES);
236        else
237            msrpHeaders.setFailureReport(FailureReport.PARTIAL);            // we always want to see 481 or 586 errors
238        if (this.headers != null) {
239            for (Header header : this.headers) {
240                if (MsrpRequest.isMimeHeader(header.getName()))
241                    msrpHeaders.getMimeHeaders().add(header);
242                else
243                    msrpHeaders.getExtensionHeaders().add(header);
244            }
245        }
246        final MsrpRequest request = new MsrpRequest(transactionId, MsrpConstants.METHOD_SEND, msrpHeaders);
247
248        // Aborted?
249        if (this.aborted) {
250            if (this.input != null)
251                msrpHeaders.setByteRange(new ByteRange(this.offset + 1, this.offset, this.size));
252            request.setAborted(true);
253            this.complete = true;
254            return request;
255        }
256
257        // No input?
258        if (this.input == null) {
259            this.complete = true;
260            this.closed = true;
261            request.setComplete(true);
262            return request;
263        }
264
265        // Read next chunk of message content and append as body
266        final ByteArrayOutputStream body = new ByteArrayOutputStream();
267        final long startingOffset = this.offset;
268        try {
269
270            // Read until buffer is full, we've read full amount, or there's no more data to read
271            while (body.size() < MAX_REQUEST_BODY_SIZE) {
272
273                // Have we read the full amount?
274                if (this.size != -1 && this.offset >= this.size) {
275                    this.complete = true;
276                    break;
277                }
278
279                // Read more data
280                final byte[] buf = new byte[MAX_REQUEST_BODY_SIZE - body.size()];
281                final int r = this.input.read(buf);
282
283                // No more data?
284                if (r < 0) {
285                    if (this.size != -1)
286                        throw new IOException("expected to read " + this.size + " bytes but only read " + this.offset);
287                    this.complete = true;
288                    break;
289                }
290
291                // Update with new data
292                body.write(buf, 0, r);
293                this.offset += r;
294            }
295        } catch (IOException e) {
296            this.log.error("I/O error reading MRSP message input, aborting message " + this.messageId, e);
297            this.aborted = true;
298        }
299        request.setBody(body.toByteArray());
300
301        // Set byte range and request flags
302        msrpHeaders.setByteRange(new ByteRange(startingOffset + 1, this.offset, this.size));
303        request.setComplete(this.complete);
304        request.setAborted(this.aborted);
305
306        // Ensure input gets closed
307        if (this.aborted || this.complete)
308            this.close();
309
310        // Done
311        return request;
312    }
313
314    @Override
315    public void remove() {
316        throw new UnsupportedOperationException();
317    }
318
319// Closeable
320
321    /**
322     * Close the {@link InputStream} associated with this instance.
323     * This method is idempotent.
324     *
325     * <p>
326     * This class ensures that this method gets invoked after the final {@link MsrpMessage} is
327     * retrieved from {@link #next}.
328     * </p>
329     *
330     * <p>
331     * If the message has not been fully sent when this method is invoked, the message will be aborted.
332     * </p>
333     */
334    @Override
335    public synchronized void close() {
336
337        // Sanity check
338        if (this.closed)
339            return;
340        this.closed = true;
341
342        // If not completed, abort
343        if (!this.complete)
344            this.aborted = true;
345
346        // Close input
347        try {
348            if (this.input != null)
349                this.input.close();
350        } catch (IOException e) {
351            // ignore
352        }
353    }
354}
