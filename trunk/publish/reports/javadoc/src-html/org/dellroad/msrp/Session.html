090    /**
091     * Close this session.
092     *
093     * <p>
094     * Any {@link FailureListener}s associated with queued messages will be notified.
095     * This method is idempotent.
096     * </p>
097     *
098     * @param cause error that occurred, if any, otherwise null
099     * @return true if this instance was closed, false if this instance was already closed
100     */
101    public boolean close(final Exception cause) {
102        synchronized (this.msrp) {
103            if (this.closed)
104                return false;
105            if (this.log.isDebugEnabled())
106                this.log.debug("closing " + this);
107            try {
108                this.flushOutputQueue();            // to ensure any failure responses get sent
109            } catch (IOException e) {
110                // ignore
111            }
112            for (OutputChunks chunks : this.outputChunks.values()) {
113                chunks.close();
114                if (chunks.hasNext()) {
115                    chunks.notifyFailure(this, this.callbackExecutor,
116                      new Status(MsrpConstants.RESPONSE_CODE_SESSION_DOES_NOT_EXIST, "Session closed"));
117                }
118            }
119            this.inputChunks.clear();
120            this.outputChunks.clear();
121            this.outputTransactions.clear();
122            this.outputQueue.clear();
123            this.closed = true;
124            this.msrp.handleSessionClosed(this);
125
126            // Notify listener
127            this.callbackExecutor.execute(new Runnable() {
128                @Override
129                public void run() {
130                    try {
131                        Session.this.listener.sessionClosed(Session.this, cause);
132                    } catch (ThreadDeath t) {
133                        throw t;
134                    } catch (Throwable t) {
135                        Session.this.log.error("error in listener notification", t);
136                    }
137                }
138            });
139            return true;
140        }
141    }

143    /**
144     * Enqueue an outgoing message specified as a {@code byte[]} array.
145     *
146     * @param content message content
147     * @param contentType content type
148     * @param headers other headers (MIME and/or extension), or null for none
149     * @param reportListener success and/or failure listener, or null for none
150     * @return unique message ID, or null if this instance is closed
151     * @throws IllegalArgumentException if {@code content} or {@code contentType} is null
152     * @throws IllegalArgumentException if {@code headers} contains an invalid header name or value
153     */
154    public String send(byte[] content, String contentType, Iterable<? extends Header> headers, ReportListener reportListener) {
155        if (content == null)
156            throw new IllegalArgumentException("null content");
157        return this.doSend(new ByteArrayInputStream(content), content.length, contentType, headers, reportListener);
158    }

195    /**
196     * Cancel an outgoing message previously sent.
197     *
198     * @param messageId message ID returned by {@code send()}
199     * @throws IllegalArgumentException if {@code messageId} is null
200     * @return true if message was canceled, false if message has already been completely sent
201     */
202    public boolean cancel(String messageId) {
203        if (messageId == null)
204            throw new IllegalArgumentException("null messageId");
205        synchronized (this.msrp) {
206            final OutputChunks chunks = this.outputChunks.get(messageId);
207            if (chunks == null)
208                return false;
209            chunks.close();
210            return chunks.isAborted();
211        }
212    }

214    /**
215     * Enqueue an outgoing success report.
216     *
217     * @param toPath path to the peer; should equal the {@code From-Path} from the received message
218     * @param messageId the {@code Message-ID} from the received message
219     * @param status success status, or null for default {@code 200 Message Delivered}
220     * @param byteRange byte range successfully received
221     * @return true if report was sent, false if report could not be sent because this instance is closed
222     * @throws IllegalArgumentException if {@code toPath} is null or empty
223     * @throws IllegalArgumentException if {@code messageId} is null or invalid
224     * @throws IllegalArgumentException if {@code byteRange} is null
225     */
226    public boolean sendSuccessReport(List<MsrpUri> toPath, String messageId, ByteRange byteRange, Status status) {
227        if (toPath == null || toPath.isEmpty() || toPath.get(0) == null)
228            throw new IllegalArgumentException("null/empty toPath");
229        if (messageId == null)
230            throw new IllegalArgumentException("null messageId");
231        if (byteRange == null)
232            throw new IllegalArgumentException("null byteRange");
233        if (status == null)
234            status = new Status(MsrpConstants.RESPONSE_CODE_OK, "Message delivered");
235        synchronized (this.msrp) {
236            if (this.closed)
237                return false;
238            this.enqueueReport(toPath, messageId, status, byteRange);
239        }
240        return true;
241    }

243    /**
244     * Enqueue an outgoing failure report.
245     *
246     * @param toPath path to the peer; should equal the {@code From-Path} from the received message
247     * @param messageId the {@code Message-ID} from the received message
248     * @param status failure status
249     * @return true if report was sent, false if report could not be sent because this instance is closed
250     * @throws IllegalArgumentException if {@code toPath} is null or empty
251     * @throws IllegalArgumentException if {@code messageId} is null or invalid
252     * @throws IllegalArgumentException if {@code status} is null
253     */
254    public boolean sendFailureReport(List<MsrpUri> toPath, String messageId, Status status) {
255        if (toPath == null || toPath.isEmpty() || toPath.get(0) == null)
256            throw new IllegalArgumentException("null/empty toPath");
257        if (messageId == null)
258            throw new IllegalArgumentException("null messageId");
259        if (status == null)
260            throw new IllegalArgumentException("null status");
261        synchronized (this.msrp) {
262            if (this.closed)
263                return false;
264            this.enqueueReport(toPath, messageId, status, null);
265        }
266        return true;
267    }

271    String doSend(InputStream input, int size, String contentType,
272      Iterable<? extends Header> headers, ReportListener reportListener) {
273        synchronized (this.msrp) {
274
275            // Sanity check
276            if (this.closed)
277                return null;
278
279            // Enqueue output message
280            final OutputChunks chunks = new OutputChunks(this.localURI,
281              this.remoteURI, input, size, contentType, headers, reportListener);
282            final String messageId = chunks.getMessageId();
283            this.outputChunks.put(messageId, chunks);
284
285            // Wakeup MSRP thread so it will invoke performHousekeeping()
286            this.msrp.wakeup();
287
288            // Done
289            return messageId;
290        }
291    }

302    // Handle an incoming {@link MsrpMessage} received on this session.
303    void handleMessage(MsrpMessage msg) {
304
305        // Sanity check
306        if (msg == null)
307            throw new IllegalArgumentException("null msg");
308        if (this.closed)
309            throw new IllegalStateException("session is closed");
310
311        // Handle request or response
312        if (msg instanceof MsrpRequest)
313            this.handleRequest((MsrpRequest)msg);
314        else if (msg instanceof MsrpResponse)
315            this.handleResponse((MsrpResponse)msg);
316        else
317            this.log.error("Session.handleInput(): ignoring unknown message of type " + msg.getClass().getName());
318    }
319
320    // Handle request
321    private void handleRequest(MsrpRequest request) {
322        switch (request.getMethod()) {
323        case MsrpConstants.METHOD_SEND:
324            this.handleSend(request);
325            break;
326        case MsrpConstants.METHOD_REPORT:
327            this.handleReport(request);
328            break;
329        default:
330            if (!FailureReport.NO.equals(request.getHeaders().getFailureReport())) {
331                this.outputQueue.add(Session.createMsrpResponse(request,
332                  MsrpConstants.RESPONSE_CODE_UNKNOWN_METHOD, "Unknown method `" + request.getMethod() + "'"));
333            }
334            return;
335        }
336    }
337
338    // Handle SEND request
339    private void handleSend(MsrpRequest request) {
340
341        // Get/create input message
342        final MsrpHeaders headers = request.getHeaders();
343        final String messageId = headers.getMessageId();
344        InputChunks chunks0 = this.inputChunks.get(messageId);
345        if (chunks0 == null) {
346            chunks0 = new InputChunks(messageId, this.msrp.getMaxContentLength());
347            this.inputChunks.put(messageId, chunks0);
348        }
349        final InputChunks chunks = chunks0;
350
351        // Process request
352        final boolean complete;
353        try {
354            complete = chunks.handleSend(request);
355        } catch (ProtocolException e) {
356            if (!FailureReport.NO.equals(headers.getFailureReport())) {
357                this.outputQueue.add(Session.createMsrpResponse(request,
358                  MsrpConstants.RESPONSE_CODE_BAD_REQUEST, "Protocol error: " + e.getMessage()));
359            }
360            this.close(e);
361            return;
362        }
363
364        // Reply to transaction
365        if (!FailureReport.NO.equals(headers.getFailureReport()) && !FailureReport.PARTIAL.equals(headers.getFailureReport()))
366            this.outputQueue.add(Session.createMsrpResponse(request, MsrpConstants.RESPONSE_CODE_OK, "OK"));
367
368        // Is message aborted?
369        if (chunks.isAborted()) {
370            this.inputChunks.remove(messageId);
371            return;
372        }
373
374        // Is message still incomplete?
375        if (!complete)
376            return;
377
378        // Remove from incoming messages
379        this.inputChunks.remove(messageId);
380
381        // Notify listener of reception of complete message
382        final byte[] content = chunks.getContent();
383        final TreeSet<Header> combinedHeaders = new TreeSet<Header>(Header.SORT_BY_NAME);
384        combinedHeaders.addAll(headers.getMimeHeaders());
385        combinedHeaders.addAll(headers.getExtensionHeaders());
386        this.callbackExecutor.execute(new Runnable() {
387            @Override
388            public void run() {
389                try {
390                    Session.this.listener.sessionReceivedMessage(Session.this, chunks.getFromPath(), messageId,
391                      content, headers.getContentType(), combinedHeaders, chunks.isSuccessReport(),
392                      FailureReport.YES.equals(chunks.getFailureReport()));
393                } catch (ThreadDeath t) {
394                    throw t;
395                } catch (Throwable t) {
396                    Session.this.log.error("error in listener notification", t);
397                }
398            }
399        });
400    }

402    // Handle REPORT request
403    private void handleReport(MsrpRequest request) {
404
405        // Extract info
406        final MsrpHeaders requestHeaders = request.getHeaders();
407        final String messageId = requestHeaders.getMessageId();
408        if (messageId == null) {
409            this.log.warn("rec'd REPORT with no message ID in session from " + this.remoteURI + ", ignoring");
410            return;
411        }
412        final Status status = requestHeaders.getStatus();
413        if (status == null) {
414            this.log.warn("rec'd REPORT with no Status in session from " + this.remoteURI + ", ignoring");
415            return;
416        }
417        final boolean success = status.getNamespace() == 0 && status.getCode() / 100 == 2;
418        final ByteRange byteRange = requestHeaders.getByteRange();
419        if (success && byteRange == null) {
420            this.log.warn("rec'd success REPORT with no ByteRange from " + this.remoteURI + ", ignoring");
421            return;
422        }
423
424        // Find outgoing message
425        final OutputChunks chunks = this.outputChunks.get(messageId);
426        if (chunks == null)
427            return;
428
429        // Notify listener
430        if (success)
431            chunks.notifySuccess(this, this.callbackExecutor, byteRange);
432        else
433            chunks.notifyFailure(this, this.callbackExecutor, status);
434    }

436    // Handle transaction response
437    private void handleResponse(MsrpResponse response) {
438
439        // Find the corresponding transaction and output chunks
440        final OutputTransaction transaction = this.outputTransactions.remove(response.getTransactionId());
441        if (transaction == null)
442            return;
443        final OutputChunks chunks = transaction.getOutputChunks();
444
445        // Success is easy :-)
446        if (response.getCode() < 300)
447            return;
448
449        // Notify failure listener, if any
450        chunks.notifyFailure(this, this.callbackExecutor, response.toStatus());
451
452        // Interrupt output chunks
453        chunks.close();
454
455        // Handle specific codes
456        switch (response.getCode()) {
457        case MsrpConstants.RESPONSE_CODE_SESSION_DOES_NOT_EXIST:
458        case MsrpConstants.RESPONSE_CODE_SESSION_ALREADY_BOUND:
459            this.close(new Exception("rec'd error from remote: " + response.getResultString()));
460            break;
461        default:
462            break;
463        }
464    }

466    /**
467     * Create a {@link MsrpResponse} corresponding to the given request with the given error code and comment.
468     */
469    static MsrpResponse createMsrpResponse(MsrpRequest request, int code, String comment) {
470
471        // Sanity check
472        if (request == null)
473            throw new IllegalArgumentException("null request");
474        new Status(code, comment);
475
476        // Build response
477        final MsrpHeaders requestHeaders = request.getHeaders();
478        final MsrpHeaders responseHeaders = new MsrpHeaders();
479        if (MsrpConstants.METHOD_SEND.equals(request.getMethod()))
480            responseHeaders.getToPath().add(requestHeaders.getFromPath().get(0));
481        else
482            responseHeaders.getToPath().addAll(requestHeaders.getFromPath());
483        responseHeaders.getFromPath().add(requestHeaders.getToPath().get(0));
484        return new MsrpResponse(request.getTransactionId(), code, comment, responseHeaders);
485    }

487    private void enqueueReport(List<MsrpUri> toPath, String messageId, Status status, ByteRange byteRange) {
488        final MsrpHeaders headers = new MsrpHeaders();
489        headers.getToPath().addAll(toPath);
490        headers.getFromPath().add(Session.this.localURI);
491        headers.setMessageId(messageId);
492        headers.setStatus(status);
493        if (byteRange != null)
494            headers.setByteRange(byteRange);
495        this.outputQueue.add(new MsrpRequest(MsrpMessage.randomId(), MsrpConstants.METHOD_REPORT, headers));
496    }

498// Housekeeping
499
500    void performHousekeeping() throws IOException {
501
502        // Create connection if we are active and none exists yet. Note: this can block doing DNS lookups XXX
503        if (this.connection == null && this.activeEndpoint != null)
504            this.connection = this.msrp.createConnection(this.activeEndpoint);
505
506        // If we don't have a connection for longer than the connect timeout, fail
507        if (this.connection == null) {
508            final long age = (System.nanoTime() - this.startTime) / 1000000L;
509            if (age >= this.msrp.getConnectTimeout())
510                throw new IOException("session not bound after " + this.msrp.getConnectTimeout() + "ms");
511        }
512
513        // Add the next chunk of each outstanding message to the output queue in round-robin fashion
514        if (this.outputQueue.isEmpty()) {
515            for (OutputChunks chunks : this.outputChunks.values()) {
516                if (chunks.hasNext()) {
517                    final MsrpMessage request = chunks.next();
518                    this.outputQueue.add(request);
519                    this.outputTransactions.put(request.getTransactionId(),
520                      new OutputTransaction(chunks, request.getTransactionId()));
521                }
522            }
523        }
524
525        // If we have a connection, move enqueued chunks from my output queue to connection's output queue
526        this.flushOutputQueue();
527
528        // Scrub output messages that are complete and either have already been reported on or have timed out waiting
529        for (Iterator<OutputChunks> i = this.outputChunks.values().iterator(); i.hasNext(); ) {
530            final OutputChunks chunks = i.next();
531            if (chunks.hasNext())
532                continue;
533            if (chunks.getReportListener() == null || chunks.getIdleTime() > MAX_MESSAGE_IDLE_TIME_MILLIS)
534                i.remove();
535        }
536
537        // Scrub orphaned input messages
538        for (Iterator<InputChunks> i = this.inputChunks.values().iterator(); i.hasNext(); ) {
539            final InputChunks chunks = i.next();
540            assert !chunks.isComplete() && !chunks.isAborted();
541            if (chunks.getIdleTime() > MAX_MESSAGE_IDLE_TIME_MILLIS) {
542                if (!FailureReport.NO.equals(chunks.getFailureReport())) {
543                    this.enqueueReport(chunks.getFromPath(), chunks.getMessageId(),
544                      new Status(MsrpConstants.RESPONSE_CODE_TIMEOUT, "Missing message chunks never arrived"), null);
545                }
546                i.remove();
547                continue;
548            }
549        }
550
551        // Scrub transactions that have timed out waiting for transaction responses
552        for (Iterator<OutputTransaction> i = this.outputTransactions.values().iterator(); i.hasNext(); ) {
553            final OutputTransaction outputTransaction = i.next();
554            if (outputTransaction.getAge() >= MAX_TRANSACTION_AGE_MILLIS) {
555                outputTransaction.getOutputChunks().notifyFailure(this, this.callbackExecutor,
556                  new Status(MsrpConstants.RESPONSE_CODE_TIMEOUT, "No response rec'd for transaction"));
557                i.remove();
558            }
559        }
560    }
561
562    // If we have a connection, move enqueued chunks from my output queue to connection's output queue
563    private void flushOutputQueue() throws IOException {
564        if (this.connection != null) {
565            for (MsrpMessage message; (message = this.outputQueue.pollFirst()) != null; )
566                this.connection.write(message);
567        }
568    }

572    /**
573     * This object is needed so that when a failure report is requested we can keep track of who to notify.
574     */
575    private class OutputTransaction {
576
577        private final OutputChunks chunks;
578        private final String transactionId;
579        private final long sendTime;
580
581        OutputTransaction(OutputChunks chunks, String transactionId) {
582            this.chunks = chunks;
583            this.transactionId = transactionId;
584            this.sendTime = System.nanoTime();
585        }
586
587        public OutputChunks getOutputChunks() {
588            return this.chunks;
589        }
590
591        public String getTransactionId() {
592            return this.transactionId;
593        }
594
595        public long getAge() {
596            return (System.nanoTime() - this.sendTime) / 1000000L;
597        }
598    }
