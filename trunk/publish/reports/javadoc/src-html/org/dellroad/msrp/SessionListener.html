008package org.dellroad.msrp;
009
010import java.util.List;
011import java.util.SortedSet;
012
013import org.dellroad.msrp.msg.Header;
014
015/**
016 * Callback interface for MSRP {@link Session}s.
017 */
018public interface SessionListener {
019
020    /**
021     * Notification that a {@link Session} has been closed or failed due to an error.
022     *
023     * @param session the session that closed
024     * @param cause the error that occurred, or null if the session was closed manually by the application
025     */
026    void sessionClosed(Session session, Exception cause);
027
028    /**
029     * Notification that an MSRP message has been received in a session.
030     *
031     * <p>
032     * Note that this method is responsible for (eventually) triggering a success or failure report
033     * depending on {@code successReport} and/or {@code failureReport}.
034     * </p>
035     *
036     * @param session the session on which the message was received
037     * @param fromPath the path that the message took to get here
038     * @param messageId sender's unique ID for this message
039     * @param content message content, or null if message contained no content
040     * @param contentType message content type, or null if message contained no content
041     * @param headers other headers (including any MIME headers) sorted by name
042     * @param successReport whether a success report is requested by the sender
043     * @param failureReport whether a failure report is requested by the sender
044     * @see Session#sendSuccessReport Session.sendSuccessReport()
045     * @see Session#sendFailureReport Session.sendFailureReport()
046     */
047    void sessionReceivedMessage(Session session, List<MsrpUri> fromPath, String messageId, byte[] content, String contentType,
048      SortedSet<Header> headers, boolean successReport, boolean failureReport);
049}
