008package org.dellroad.msrp.msg;
009
010import java.io.ByteArrayOutputStream;
011import java.util.regex.Pattern;
012
013import org.dellroad.msrp.MsrpConstants;
014
015/**
016 * Parses body content bytes separated by MSRP end-line boundaries.
017 *
018 * <p>
019 * Instances are not thread safe.
020 * </p>
021 */
022public class BoundaryInputParser {
023
024    private final byte[] terminator;        // terminator string (with variable flag byte)
025    private final int flagOffset;           // offset of flag byte in terminator
026
027    private boolean boundary = true;        // whether we are between body parts
028    private int matched;                    // how many bytes of terminator we have matched so far
029
030    /**
031     * Constructor.
032     *
033     * @param transactionId transaction ID
034     * @throws IllegalArgumentException if {@code transactionId} is null or invalid
035     */
036    public BoundaryInputParser(String transactionId) {
037
038        // Sanity check
039        if (transactionId == null)
040            throw new IllegalArgumentException("null transactionId");
041        if (!Pattern.compile(Util.IDENT_REGEX).matcher(transactionId).matches())
042            throw new IllegalArgumentException("illegal transactionId");
043
044        // Build byte sequence that ends the body (note, the '?' is a placeholder for the flag byte)
045        this.terminator = (Util.CRLF + MsrpConstants.END_LINE_PREFIX + transactionId + "?" + Util.CRLF).getBytes(Util.UTF8);
046        this.flagOffset = 2 + MsrpConstants.END_LINE_PREFIX.length() + transactionId.length();
047        assert terminator[flagOffset] == (byte)'?';
048    }
049
050    /**
051     * Input the next byte and get back any new content bytes that have been determined to not be part of the boundary string.
052     *
053     * @param b input byte
054     * @return zero or more additional body bytes, or null if the boundary string has been seen
055     * @throws ProtocolException if the body exceeds the maximum allowed length
056     */
057    public byte[] inputContentByte(byte b) {
058
059        // No longer at boundary
060        this.boundary = false;
061
062        // Does this byte match the next terminator prefix byte? Note special case for flag byte.
063        if (matched == this.flagOffset ?
064          b == MsrpConstants.FLAG_INCOMPLETE || b == MsrpConstants.FLAG_COMPLETE || b == MsrpConstants.FLAG_ABORT :
065          b == this.terminator[matched]) {
066
067            // Replace '?' with the actual flag seen when we see it
068            if (this.matched == this.flagOffset)
069                this.terminator[this.flagOffset] = b;
070
071            // Increment the number of terminator bytes matched; have we now matched the entire terminator?
072            if (++this.matched == this.terminator.length) {
073                this.reset();
074                return null;
075            }
076
077            // End line not fully matched yet, and there is no additional data at this point
078            return new byte[0];
079        }
080
081        // Were any bytes previously matched? If not, this is easy.
082        if (this.matched == 0)
083            return new byte[] { b };
084
085        // Match failure: roll back to the next longest prefix (this is O(n^2)... better would be Knuth-Morris-Pratt variant)
086        final ByteArrayOutputStream buf = new ByteArrayOutputStream(this.terminator.length);
087        buf.write(this.terminator[0]);
088    retractionLoop:
089        for (int offset = 1; this.matched > 0; offset++) {
090
091            // Check whether terminator[offset..end] + b matches a prefix
092            for (int i = 0; i < this.matched; i++) {
093                final byte next = (i < this.matched - 1) ? terminator[offset + i] : b;
094                if (terminator[i] != next) {
095                    buf.write(--this.matched > 0 ? this.terminator[offset] : b);
096                    continue retractionLoop;
097                }
098            }
099            break;
100        }
101
102        // Return available bytes
103        return buf.toByteArray();
104    }
105
106    /**
107     * Determine whether this instance is sitting at a content boundary.
108     *
109     * <p>
110     * This will be true after initial construction, invocation of {@link #reset},
111     * or an invocation {@link #inputContentByte inputContentByte()} that returned a null value.
112     * </p>
113     */
114    public boolean isOnBoundary() {
115        return this.boundary;
116    }
117
118    /**
119     * Reset parse state.
120     */
121    public void reset() {
122        this.boundary = true;
123        this.matched = 0;
124    }
125
126    /**
127     * Get the flag byte that was found in the boundary string.
128     *
129     * <p>
130     * If the complete boundary string has not yet been encountered, then the return
131     * value from this method is undefined.
132     * </p>
133     */
134    public byte getFlagByte() {
135        return this.terminator[this.flagOffset];
136    }
137}
