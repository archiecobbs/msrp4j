008package org.dellroad.msrp.msg;
009
010import java.io.EOFException;
011import java.io.FilterInputStream;
012import java.io.IOException;
013import java.io.InputStream;
014
015/**
016 * Reads from an underlying {@link InputStream} until an MSRP end-line boundary is encountered.
017 */
018public class BoundaryInputStream extends FilterInputStream {
019
020    private final BoundaryInputParser parser;
021    private byte[] data;
022    private int offset;
023    private boolean eof;
024
025    /**
026     * Constructor.
027     *
028     * @param input underlying input
029     * @param transactionId transaction ID
030     * @throws IllegalArgumentException if {@code input} is null
031     * @throws IllegalArgumentException if {@code transactionId} is null or invalid
032     */
033    public BoundaryInputStream(InputStream input, String transactionId) {
034        super(input);
035        this.parser = new BoundaryInputParser(transactionId);
036    }
037
038// InputStream methods
039
040    /**
041     * Read the next byte.
042     *
043     * @return next byte prior to boundary, or -1 if the boundary has been seen
044     * @throws EOFException if EOF is detected before the boundary is seen
045     * @throws IOException if an I/O error occurs
046     */
047    @Override
048    public int read() throws IOException {
049
050        // Done?
051        if (this.eof)
052            return -1;
053
054        // Shift input bytes into parser until we get something back
055        while (this.data == null || this.offset == this.data.length) {
056            final int r = super.read();
057            if (r == -1)
058                throw new EOFException("detected EOF before boundary string was matched");
059            if ((this.data = this.parser.inputContentByte((byte)r)) == null) {
060                this.eof = true;
061                return -1;
062            }
063            this.offset = 0;
064        }
065
066        // Return next byte
067        return this.data[this.offset++];
068    }
069
070    @Override
071    public int read(byte[] b, int off, int len) throws IOException {
072        int total = 0;
073        while (off < len) {
074            final int r = this.read();
075            if (r == -1)
076                return total == 0 ? -1 : total;
077            b[off++] = (byte)r;
078            total++;
079        }
080        return total;
081    }
082
083    @Override
084    public long skip(long n) throws IOException {
085        long total = 0;
086        while (n-- > 0) {
087            if (this.read() == -1)
088                break;
089            total++;
090        }
091        return total;
092    }
093
094    @Override
095    public int available() throws IOException {
096        return this.data != null ? this.data.length - this.offset : 0;
097    }
098
099    /**
100     * Mark this instance. This operation is not supported by {@link BoundaryInputStream}.
101     *
102     * @throws UnsupportedOperationException always
103     */
104    @Override
105    public void mark(int readlimit) {
106        throw new UnsupportedOperationException();
107    }
108
109    /**
110     * Reset this instance. This operation is not supported by {@link BoundaryInputStream}.
111     *
112     * @throws UnsupportedOperationException always
113     */
114    @Override
115    public void reset() {
116        throw new UnsupportedOperationException();
117    }
118
119    /**
120     * Determine if mark/reset is supported by this instance. Mark/reset is not supported by {@link BoundaryInputStream}.
121     *
122     * @return false always
123     */
124    @Override
125    public boolean markSupported() {
126        return false;
127    }
128
129// Other public methods
130
131    /**
132     * Get the flag byte that was found in the boundary string.
133     *
134     * <p>
135     * If this instance's {@link #read read()} method has not yet returned -1, then the return
136     * value from this method is undefined.
137     * </p>
138     */
139    public byte getFlagByte() {
140        return this.parser.getFlagByte();
141    }
142}
