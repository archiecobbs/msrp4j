008package org.dellroad.msrp.msg;
009
010import java.util.regex.Matcher;
011import java.util.regex.Pattern;
012
013/**
014 * MSRP {@code Byte-Range} header value.
015 *
016 * <p>
017 * Instances are immutable.
018 * </p>
019 */
020public class ByteRange {
021
022    /**
023     * Used for empty messages: {@code 1-0/0}.
024     */
025    public static final ByteRange EMPTY = new ByteRange(1, 0, 0);
026
027    /**
028     * Used for indeterminate "all" byte range: {@code 1-*}{@code /*}.
029     */
030    public static final ByteRange ALL = new ByteRange(1, -1, -1);
031
032    private static final String BYTE_RANGE_REGEX = "([0-9]+)-(([0-9]+)|\\*)/(([0-9]+)|\\*)";
033
034    private final long start;
035    private final long end;
036    private final long total;
037
038    /**
039     * Constructs an instance indicating the entire content with the specified length.
040     * Equivalent to {@code ByteRange(1, total, total)}.
041     *
042     * @param total total number of bytes, or -1 if unspecified
043     * @throws IllegalArgumentException if {@code total} is less than -1
044     */
045    public ByteRange(long total) {
046        this(1, total, total);
047    }
048
049    /**
050     * Constructor.
051     *
052     * @param start start byte position (1-based)
053     * @param end end byte position (1-based), or -1 if unspecified
054     * @param total total number of bytes, or -1 if unspecified
055     * @throws IllegalArgumentException if {@code start} is less than 1
056     * @throws IllegalArgumentException if {@code end} or {@code total} is less than -1
057     * @throws IllegalArgumentException if {@code end} is not -1 and {@code start - 1} is greater than {@code end}
058     * @throws IllegalArgumentException if {@code total} is not -1 and {@code end} is greater than {@code total}
059     */
060    public ByteRange(long start, long end, long total) {
061        if (start < 1)
062            throw new IllegalArgumentException("start < 1");
063        if (end < -1)
064            throw new IllegalArgumentException("end < -1");
065        if (total < -1)
066            throw new IllegalArgumentException("total < -1");
067        if (end != -1 && start - 1 > end)
068            throw new IllegalArgumentException("start - 1 > end");
069        if (total != -1 && end > total)
070            throw new IllegalArgumentException("end > total");
071        this.start = start;
072        this.end = end;
073        this.total = total;
074    }
075
076    /**
077     * Get the start byte position.
078     *
079     * @return start byte position, at least one
080     */
081    public long getStart() {
082        return this.start;
083    }
084
085    /**
086     * Get the end byte position.
087     *
088     * @return end byte position, or -1 if not specified
089     */
090    public long getEnd() {
091        return this.end;
092    }
093
094    /**
095     * Get the total number of bytes.
096     *
097     * @return total number of bytes, or -1 if not specified
098     */
099    public long getTotal() {
100        return this.total;
101    }
102
103    /**
104     * Create an instance by parsing a {@link String}.
105     *
106     * @param string byte range expressed as a string
107     * @throws IllegalArgumentException if {@code string} is null or invalid
108     */
109    public static ByteRange fromString(String string) {
110        if (string == null)
111            throw new IllegalArgumentException("null string");
112        final Matcher matcher = Pattern.compile(BYTE_RANGE_REGEX).matcher(string);
113        if (!matcher.matches())
114            throw new IllegalArgumentException("invalid byte range " + Util.quotrunc(string));
115        try {
116            final long start = Long.parseLong(matcher.group(1), 10);
117            final long end = !matcher.group(2).equals("*") ? Long.parseLong(matcher.group(3), 10) : -1;
118            final long total = !matcher.group(4).equals("*") ? Long.parseLong(matcher.group(5), 10) : -1;
119            return new ByteRange(start, end, total);
120        } catch (IllegalArgumentException e) {
121            throw new IllegalArgumentException("invalid byte range " + Util.quotrunc(string));
122        }
123    }
124
125// Object
126
127    @Override
128    public String toString() {
129        return this.start + "-" + (this.end != -1 ? this.end : "*") + "/" + (this.total != -1 ? this.total : "*");
130    }
131
132    @Override
133    public boolean equals(Object obj) {
134        if (obj == this)
135            return true;
136        if (obj == null || obj.getClass() != this.getClass())
137            return false;
138        final ByteRange that = (ByteRange)obj;
139        return this.start == that.start && this.end == that.end && this.total == that.total;
140    }
141
142    @Override
143    public int hashCode() {
144        return (int)((this.start << 22) ^ (this.end << 11) ^ this.total);
145    }
146}
