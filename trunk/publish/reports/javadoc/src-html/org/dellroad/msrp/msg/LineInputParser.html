008package org.dellroad.msrp.msg;
009
010import java.io.ByteArrayOutputStream;
011
012/**
013 * Parses CRLF-terminated lines.
014 */
015public class LineInputParser {
016
017    private final int maxLength;
018    private final ByteArrayOutputStream buf = new ByteArrayOutputStream();
019
020    private boolean pendingCR;
021
022    /**
023     * Constructor.
024     *
025     * @param maxLength maximum allowed line length in bytes
026     * @throws IllegalArgumentException if {@code maxLength} negative
027     */
028    public LineInputParser(int maxLength) {
029        if (maxLength < 0)
030            throw new IllegalArgumentException("maxLength < 0");
031        this.maxLength = maxLength;
032    }
033
034    /**
035     * Input the next byte.
036     *
037     * @param b input byte
038     * @return complete line (not including CRLF), or null if more bytes are needed
039     * @throws ProtocolException if the line exceeds the maximum allowed length
040     */
041    public String inputLineByte(byte b) throws ProtocolException {
042
043        // Update state machine
044        if (this.pendingCR) {
045            if (b == '\n') {
046                final String line = new String(this.buf.toByteArray(), Util.UTF8);
047                this.reset();
048                return line;
049            }
050            buf.write('\r');            // flush the pending CR
051        }
052        this.pendingCR = b == '\r';
053        if (!this.pendingCR)
054            buf.write(b);
055
056        // Check max length
057        if (buf.size() >= this.maxLength) {
058            final String prefix = new String(buf.toByteArray(), 0, Util.TRUNCATE_LENGTH, Util.UTF8);
059            throw new ProtocolException("message line starting with " + Util.quotrunc(prefix)
060              + " is too long (longer than " + this.maxLength + " bytes)");
061        }
062
063        // Done
064        return null;
065    }
066
067    /**
068     * Determine whether this instance is sitting at a line boundary.
069     *
070     * <p>
071     * This will be true after initial construction, invocation of {@link #reset},
072     * or an invocation {@link #inputLineByte inputLineByte()} that returned a non-null value.
073     * </p>
074     */
075    public boolean isBetweenLines() {
076        return this.buf.size() == 0 && !this.pendingCR;
077    }
078
079    /**
080     * Reset parse state.
081     */
082    public void reset() {
083        this.buf.reset();
084        this.pendingCR = false;
085    }
086}
