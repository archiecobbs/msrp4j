008package org.dellroad.msrp.msg;
009
010import java.util.ArrayList;
011import java.util.List;
012import java.util.SortedSet;
013import java.util.TreeSet;
014
015import org.dellroad.msrp.MsrpConstants;
016import org.dellroad.msrp.MsrpUri;
017
018/**
019 * MSRP headers.
020 */
021public class MsrpHeaders {
022
023    private final ArrayList<MsrpUri> toPath = new ArrayList<>();
024    private final ArrayList<MsrpUri> fromPath = new ArrayList<>();
025    private String messageId;
026    private boolean successReport;
027    private FailureReport failureReport = FailureReport.YES;
028    private ByteRange byteRange;
029    private Status status;
030    private String contentType;
031    private final TreeSet<Header> extensionHeaders = new TreeSet<>(Header.SORT_BY_NAME);
032    private final TreeSet<Header> mimeHeaders = new TreeSet<>(Header.SORT_BY_NAME);
033
034    public List<MsrpUri> getToPath() {
035        return this.toPath;
036    }
037
038    public List<MsrpUri> getFromPath() {
039        return this.fromPath;
040    }
041
042    public String getMessageId() {
043        return this.messageId;
044    }
045    public void setMessageId(String messageId) {
046        this.messageId = messageId;
047    }
048
049    public boolean isSuccessReport() {
050        return this.successReport;
051    }
052    public void setSuccessReport(boolean successReport) {
053        this.successReport = successReport;
054    }
055
056    public FailureReport getFailureReport() {
057        return this.failureReport;
058    }
059    public void setFailureReport(FailureReport failureReport) {
060        this.failureReport = failureReport;
061    }
062
063    public ByteRange getByteRange() {
064        return this.byteRange;
065    }
066    public void setByteRange(ByteRange byteRange) {
067        this.byteRange = byteRange;
068    }
069
070    public Status getStatus() {
071        return this.status;
072    }
073    public void setStatus(Status status) {
074        this.status = status;
075    }
076
077    public String getContentType() {
078        return this.contentType;
079    }
080    public void setContentType(String contentType) {
081        this.contentType = contentType;
082    }
083
084    /**
085     * MIME headers, sorted by header name case-insensitively.
086     */
087    public SortedSet<Header> getMimeHeaders() {
088        return this.mimeHeaders;
089    }
090
091    /**
092     * Extension headers, sorted by header name case-insensitively.
093     */
094    public SortedSet<Header> getExtensionHeaders() {
095        return this.extensionHeaders;
096    }
097
098// Object
099
100    @Override
101    public boolean equals(Object obj) {
102        if (obj == this)
103            return true;
104        if (obj == null || obj.getClass() != this.getClass())
105            return false;
106        final MsrpHeaders that = (MsrpHeaders)obj;
107        if (!this.toPath.equals(that.toPath))
108            return false;
109        if (!this.fromPath.equals(that.fromPath))
110            return false;
111        if (!(this.messageId != null ? this.messageId.equals(that.messageId) : that.messageId == null))
112            return false;
113        if (this.successReport != that.successReport)
114            return false;
115        if (this.failureReport != that.failureReport)
116            return false;
117        if (!(this.byteRange != null ? this.byteRange.equals(that.byteRange) : that.byteRange == null))
118            return false;
119        if (!(this.status != null ? this.status.equals(that.status) : that.status == null))
120            return false;
121        if (!(this.contentType != null ? this.contentType.equals(that.contentType) : that.contentType == null))
122            return false;
123        if (!this.extensionHeaders.equals(that.extensionHeaders))
124            return false;
125        if (!this.mimeHeaders.equals(that.mimeHeaders))
126            return false;
127        return true;
128    }
129
130    @Override
131    public int hashCode() {
132        return this.toPath.hashCode()
133          ^ this.fromPath.hashCode()
134          ^ (this.messageId != null ? this.messageId.hashCode() : 0)
135          ^ (this.successReport ? 1 : 0)
136          ^ (this.failureReport != null ?  this.failureReport.hashCode() : 0)
137          ^ (this.byteRange != null ? this.byteRange.hashCode() : 0)
138          ^ (this.status != null ? this.status.hashCode() : 0)
139          ^ (this.contentType != null ? this.contentType.hashCode() : 0)
140          ^ this.extensionHeaders.hashCode()
141          ^ this.mimeHeaders.hashCode();
142    }
143
144    @Override
145    public String toString() {
146        final StringBuilder buf = new StringBuilder();
147        if (!this.toPath.isEmpty()) {
148            buf.append(MsrpConstants.TO_PATH_HEADER).append(": ");
149            for (MsrpUri uri : this.toPath)
150                buf.append(uri);
151            buf.append(Util.CRLF);
152        }
153        if (!this.toPath.isEmpty()) {
154            buf.append(MsrpConstants.FROM_PATH_HEADER).append(": ");
155            for (MsrpUri uri : this.fromPath)
156                buf.append(uri);
157            buf.append(Util.CRLF);
158        }
159        if (this.messageId != null)
160            buf.append(MsrpConstants.MESSAGE_ID_HEADER).append(": ").append(this.messageId).append(Util.CRLF);
161        if (this.successReport)
162            buf.append(MsrpConstants.SUCCESS_REPORT_HEADER).append(": ").append(MsrpConstants.YES_HEADER_VALUE).append(Util.CRLF);
163        if (this.failureReport != null && this.failureReport != FailureReport.YES)
164            buf.append(MsrpConstants.FAILURE_REPORT_HEADER).append(": ").append(this.failureReport).append(Util.CRLF);
165        if (this.byteRange != null)
166            buf.append(MsrpConstants.BYTE_RANGE_HEADER).append(": ").append(this.byteRange).append(Util.CRLF);
167        if (this.status != null)
168            buf.append(MsrpConstants.STATUS_HEADER).append(": ").append(this.status).append(Util.CRLF);
169        for (Header header : this.extensionHeaders)
170            buf.append(header).append(Util.CRLF);
171        for (Header header : this.mimeHeaders)
172            buf.append(header).append(Util.CRLF);
173        if (this.contentType != null)
174            buf.append(MsrpConstants.CONTENT_TYPE_HEADER).append(": ").append(this.contentType).append(Util.CRLF);
175        return buf.toString();
176    }
177}
