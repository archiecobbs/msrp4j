008package org.dellroad.msrp.msg;

010import java.io.ByteArrayOutputStream;
011import java.net.URISyntaxException;
012import java.util.List;
013import java.util.regex.Matcher;
014import java.util.regex.Pattern;

016import org.dellroad.msrp.MsrpConstants;

019/**
020 * Stateful MSRP message parser.
021 *
022 * <p>
023 * Instances are configured with various size limits to avoid unbounded memory usage.
024 * </p>
025 */
026public class MsrpInputParser {
027
028    /** Default maximum line length in bytes ({@value #DEFAULT_MAX_LINE_LENGTH}) */
029    public static final int DEFAULT_MAX_LINE_LENGTH = 16384;               // 16K
030
031    /** Default maximum content length ({@value #DEFAULT_MAX_CONTENT_LENGTH}) */
032    public static final int DEFAULT_MAX_CONTENT_LENGTH = 16 * 1024 * 1024;  // 16M
033
034    /** Default maximum number of URIs in {@code To-Path} or {@code From-Path} ({@value #DEFAULT_MAX_PATH_URIS}) */
035    public static final int DEFAULT_MAX_PATH_URIS = 32;
036
037    /** Default maximum number of MIME headers length ({@value #DEFAULT_MAX_MIME_HEADERS}) */
038    public static final int DEFAULT_MAX_MIME_HEADERS = 16;
039
040    /** Default maximum number of extension headers ({@value #DEFAULT_MAX_EXTENSION_HEADERS}) */
041    public static final int DEFAULT_MAX_EXTENSION_HEADERS = 32;
042
043    private static final Pattern REQUEST_LINE_PATTERN = Pattern.compile("MSRP ([-.+%=\\p{Alnum}]{3,31}) ([A-Z]+)");
044    private static final Pattern RESPONSE_LINE_PATTERN = Pattern.compile("MSRP ([-.+%=\\p{Alnum}]{3,31}) ([0-9]{3})( (.*))?");
045
046    private static final Header HEADER_EOF = new Header("dummy", "dummy");
047
048    private final LineInputParser lineParser;
049    private final int maxBodySize;
050    private final int maxPathUris;
051    private final int maxMimeHeaders;
052    private final int maxExtensionHeaders;
053
054    private State state = State.FIRST_LINE;     // parse state
055    private MsrpMessage message;                // the message we are building
056    private String endLine;
057    private boolean allowBody;
058    private ByteArrayOutputStream body;
059    private BoundaryInputParser boundaryInputParser;
060
061    /**
062     * Constructor. Uses default size limits.
063     */
064    public MsrpInputParser() {
065        this(DEFAULT_MAX_LINE_LENGTH, DEFAULT_MAX_CONTENT_LENGTH,
066          DEFAULT_MAX_PATH_URIS, DEFAULT_MAX_MIME_HEADERS, DEFAULT_MAX_EXTENSION_HEADERS);
067    }
068
069    /**
070     * Primary constructor.
071     *
072     * @param maxLineLength maximum allowed header line length in bytes
073     * @param maxBodySize maximum allowed body size in bytes
074     * @param maxPathUris maximum number of URI's allowed in {@code To-Path} or {@code From-Path} header
075     * @param maxMimeHeaders maximum number of allowed MIME headers
076     * @param maxExtensionHeaders maximum number of allowed extension headers
077     * @throws IllegalArgumentException if {@code input} is null
078     */
079    public MsrpInputParser(int maxLineLength, int maxBodySize, int maxPathUris, int maxMimeHeaders, int maxExtensionHeaders) {
080        this.lineParser = new LineInputParser(maxLineLength);
081        this.maxBodySize = maxBodySize;
082        this.maxPathUris = maxPathUris;
083        this.maxMimeHeaders = maxMimeHeaders;
084        this.maxExtensionHeaders = maxExtensionHeaders;
085    }
086
087    /**
088     * Input the next byte.
089     *
090     * @param b input byte
091     * @return next complete message, or null if more bytes are needed
092     * @throws ProtocolException if a protocol violation is detected
093     */
094    public MsrpMessage inputMessageByte(byte b) throws ProtocolException {
095
096        // Input byte
097        boolean complete = false;
098        switch (this.state) {
099        case FIRST_LINE:
100            this.inputFirstLineByte(b);
101            break;
102        case TO_PATH:
103            if (this.inputPathHeaderByte(this.message.getHeaders().getToPath(), MsrpConstants.TO_PATH_HEADER, b))
104                this.state = State.FROM_PATH;
105            break;
106        case FROM_PATH:
107            if (this.inputPathHeaderByte(this.message.getHeaders().getFromPath(), MsrpConstants.FROM_PATH_HEADER, b))
108                this.state = State.HEADER;
109            break;
110        case HEADER:
111            complete = this.inputHeaderByte(b);
112            break;
113        case BODY_CONTENT:
114            complete = this.inputBodyContentByte(b);
115            break;
116        default:
117            throw new RuntimeException("internal error");
118        }
119
120        // Message complete?
121        if (complete) {
122            final MsrpMessage result = this.message;
123            this.reset();
124            return result;
125        }
126        return null;
127    }
128
129    /**
130     * Reset parse state.
131     */
132    public void reset() {
133        this.lineParser.reset();
134        this.state = State.FIRST_LINE;
135        this.message = null;
136        this.endLine = null;
137        this.allowBody = false;
138        this.body = null;
139        this.boundaryInputParser = null;
140    }
141
142    /**
143     * Determine whether this instance is sitting at a message boundary.
144     *
145     * <p>
146     * This will be true after initial construction, an invocation of {@link #reset},
147     * or an invocation {@link #inputMessageByte inputMessageByte()} that returned a non-null value.
148     * </p>
149     */
150    public boolean isBetweenMessages() {
151        return this.state == State.FIRST_LINE && this.lineParser.isBetweenLines();
152    }
153
154// Internal parsing methods
155
156    private void inputFirstLineByte(byte b) throws ProtocolException {
157
158        // Read complete header line
159        final String line = this.lineParser.inputLineByte(b);
160        if (line == null)
161            return;
162
163        // Request or response?
164        Matcher matcher;
165        if ((matcher = REQUEST_LINE_PATTERN.matcher(line)).matches())
166            this.message = new MsrpRequest(matcher.group(1), matcher.group(2), null);
167        else if ((matcher = RESPONSE_LINE_PATTERN.matcher(line)).matches())
168            this.message = new MsrpResponse(matcher.group(1), Integer.parseInt(matcher.group(2), 10), matcher.group(4), null);
169        else
170            throw new ProtocolException("invalid start line " + Util.quotrunc(line));
171
172        // Prepare for parsing headers
173        this.endLine = MsrpConstants.END_LINE_PREFIX + this.message.getTransactionId();
174        this.allowBody = this.message instanceof MsrpRequest;
175
176        // Update state
177        this.state = State.TO_PATH;
178    }
179
180    private boolean inputHeaderByte(byte b) throws ProtocolException {
181
182        // Read complete header
183        final Header header = this.inputHeaderByteForHeader(b);
184        if (header == null)
185            return false;
186
187        // No more headers?
188        if (header == HEADER_EOF) {
189
190            // Is message allowed to have a body?
191            if (this.body != null && !this.allowBody)
192                throw new ProtocolException("message must not contain a body but does");
193
194            // Sanity check presence of MIME headers vs. having a body
195            if (this.body != null && this.message.getHeaders().getContentType() == null) {
196                throw new ProtocolException("missing header `" + MsrpConstants.CONTENT_TYPE_HEADER
197                  + "' required when message has a body");
198            }
199            if (this.body == null
200              && (this.message.getHeaders().getContentType() != null || !this.message.getHeaders().getMimeHeaders().isEmpty()))
201                throw new ProtocolException("MIME headers are not allowed when message has no body");
202
203            // If there is no body, we're done
204            if (this.body == null)
205                return true;
206
207            // Start parsing body
208            this.boundaryInputParser = new BoundaryInputParser(this.message.getTransactionId());
209            this.state = State.BODY_CONTENT;
210            return false;
211        }
212
213        // Handle header
214        final String name = header.getName();
215        final String value = header.getValue();
216        if (name.equalsIgnoreCase(MsrpConstants.MESSAGE_ID_HEADER)) {
217            if (!Pattern.compile(Util.IDENT_REGEX).matcher(value).matches())
218                throw new ProtocolException("invalid `" + name + "' header value " + Util.quotrunc(value));
219            this.message.getHeaders().setMessageId(value);
220        } else if (name.equalsIgnoreCase(MsrpConstants.SUCCESS_REPORT_HEADER)) {
221            switch (value) {
222            case MsrpConstants.YES_HEADER_VALUE:
223                this.message.getHeaders().setSuccessReport(true);
224                break;
225            case MsrpConstants.NO_HEADER_VALUE:
226                this.message.getHeaders().setSuccessReport(false);
227                break;
228            default:
229                throw new ProtocolException("invalid `" + name + "' header value " + Util.quotrunc(value));
230            }
231        } else if (name.equalsIgnoreCase(MsrpConstants.FAILURE_REPORT_HEADER)) {
232            switch (value) {
233            case MsrpConstants.YES_HEADER_VALUE:
234                this.message.getHeaders().setFailureReport(FailureReport.YES);
235                break;
236            case MsrpConstants.NO_HEADER_VALUE:
237                this.message.getHeaders().setFailureReport(FailureReport.NO);
238                break;
239            case MsrpConstants.PARTIAL_HEADER_VALUE:
240                this.message.getHeaders().setFailureReport(FailureReport.PARTIAL);
241                break;
242            default:
243                throw new ProtocolException("invalid `" + name + "' header value " + Util.quotrunc(value));
244            }
245        } else if (name.equalsIgnoreCase(MsrpConstants.BYTE_RANGE_HEADER)) {
246            try {
247                this.message.getHeaders().setByteRange(ByteRange.fromString(value));
248            } catch (IllegalArgumentException e) {
249                throw new ProtocolException("invalid `" + name + "' header value " + Util.quotrunc(value));
250            }
251        } else if (name.equalsIgnoreCase(MsrpConstants.STATUS_HEADER)) {
252            try {
253                this.message.getHeaders().setStatus(Status.fromString(value));
254            } catch (IllegalArgumentException e) {
255                throw new ProtocolException("invalid `" + name + "' header value " + Util.quotrunc(value));
256            }
257        } else if (name.equalsIgnoreCase(MsrpConstants.CONTENT_TYPE_HEADER))
258            this.message.getHeaders().setContentType(value);
259        else {
260            if (MsrpRequest.isMimeHeader(name)) {
261                this.message.getHeaders().getMimeHeaders().remove(header);               // ensure last one wins
262                if (this.message.getHeaders().getMimeHeaders().size() >= this.maxMimeHeaders)
263                    throw new ProtocolException("too many MIME headers (maximum " + this.maxMimeHeaders + ")");
264                this.message.getHeaders().getMimeHeaders().add(header);
265            } else {
266                this.message.getHeaders().getExtensionHeaders().remove(header);          // ensure last one wins
267                if (this.message.getHeaders().getExtensionHeaders().size() >= this.maxExtensionHeaders)
268                    throw new ProtocolException("too many extension headers (maximum " + this.maxExtensionHeaders + ")");
269                this.message.getHeaders().getExtensionHeaders().add(header);
270            }
271        }
272
273        // Done
274        return false;
275    }

277    // Input body content byte, returning true if message is complete
278    private boolean inputBodyContentByte(byte b) throws ProtocolException {
279
280        // Input body byte
281        final byte[] data = this.boundaryInputParser.inputContentByte(b);
282        if (data != null) {
283            this.body.write(data, 0, data.length);
284            if (this.body.size() > this.maxBodySize)
285                throw new ProtocolException("body size exceeds maximum size limit (" + this.maxBodySize + " bytes)");
286            return false;
287        }
288
289        // Add body to message and set flags
290        final MsrpRequest request = (MsrpRequest)this.message;
291        request.setBody(this.body.toByteArray());
292        switch (this.boundaryInputParser.getFlagByte()) {
293        case MsrpConstants.FLAG_INCOMPLETE:
294            request.setComplete(false);
295            request.setAborted(false);
296            break;
297        case MsrpConstants.FLAG_COMPLETE:
298            request.setComplete(true);
299            break;
300        case MsrpConstants.FLAG_ABORT:
301            request.setAborted(true);
302            break;
303        default:
304            throw new RuntimeException("internal error");
305        }
306
307        // Done
308        return true;
309    }

311    // Input required path header byte, returning true if header is complete
312    private boolean inputPathHeaderByte(List<MsrpUri> uriList, String name, byte b) throws ProtocolException {
313
314        // Get complete path(s)
315        final String paths = this.inputRequiredHeaderByte(name, b);
316        if (paths == null)
317            return false;
318
319        // Must be at least one
320        if (paths.length() == 0)
321            throw new ProtocolException("invalid empty `" + name + "' header");
322
323        // Parse path into URI's
324        int end;
325        for (int start = 0; start < paths.length(); start = end) {
326            if ((end = paths.indexOf(' ', start)) == -1)
327                end = paths.length();
328            final String uri = paths.substring(start, end);
329            if (uriList.size() >= this.maxPathUris)
330                throw new ProtocolException("too many URI's in `" + name + "' header (maximum " + this.maxPathUris + ")");
331            try {
332                uriList.add(new MsrpUri(uri));
333            } catch (URISyntaxException e) {
334                throw new ProtocolException("invalid URI " + Util.quotrunc(uri) + " in `" + name + "' header", e);
335            }
336        }
337
338        // Done
339        return true;
340    }
341
342    // Input required header byte, returning header value if header is complete
343    private String inputRequiredHeaderByte(String name, byte b) throws ProtocolException {
344        final Header header = this.inputHeaderByteForHeader(b);
345        if (header == null)
346            return null;
347        if (header == HEADER_EOF)
348            throw new ProtocolException("missing required `" + name + "' header");
349        if (!header.getName().equalsIgnoreCase(name)) {
350            throw new ProtocolException("expected required `" + name + "' header but found "
351              + Util.quotrunc(header.getName()) + " header instead");
352        }
353        return header.getValue();
354    }
355
356    // Input header byte, returning null if incomplete, header if complete, or HEADER_EOF if none remain (and setting this.body)
357    private Header inputHeaderByteForHeader(byte b) throws ProtocolException {
358
359        // Read complete header line
360        final String line = this.lineParser.inputLineByte(b);
361        if (line == null)
362            return null;
363
364        // End line?
365        assert this.endLine != null;
366        if (line.startsWith(this.endLine) && line.length() == this.endLine.length() + 1) {
367            final char flag = line.charAt(this.endLine.length());
368            if (flag != (char)MsrpConstants.FLAG_COMPLETE)
369                throw new ProtocolException("invalid end-line flag byte `" + flag + "' in message without body");
370            this.body = null;
371            return HEADER_EOF;
372        }
373
374        // Blank line?
375        if (line.length() == 0) {
376            this.body = new ByteArrayOutputStream();
377            return HEADER_EOF;
378        }
379
380        // Parse header
381        final Matcher matcher = Pattern.compile(Util.HEADER_REGEX).matcher(line);
382        if (!matcher.matches())
383            throw new ProtocolException("invalid header line " + Util.quotrunc(line));
384        return new Header(matcher.group(1), matcher.group(2));
385    }
386
387// Parse states
388
389    private enum State {
390        FIRST_LINE,
391        TO_PATH,
392        FROM_PATH,
393        HEADER,
394        BODY_CONTENT;
395    }
