008package org.dellroad.msrp.msg;
009
010import java.io.EOFException;
011import java.io.FilterInputStream;
012import java.io.IOException;
013import java.io.InputStream;
014
015/**
016 * Wraps an {@link InputStream} and parses MSRP messages from the underlying input.
017 */
018public class MsrpInputStream extends FilterInputStream {
019
020    private final MsrpInputParser parser;
021
022    /**
023     * Constructor. Uses a {@link MsrpInputParser} with default parameters.
024     *
025     * @param input underlying input
026     * @throws IllegalArgumentException if {@code input} is null
027     */
028    public MsrpInputStream(InputStream input) {
029        this(input, new MsrpInputParser());
030    }
031
032    /**
033     * Primary constructor.
034     *
035     * @param input underlying input
036     * @param parser parser
037     * @throws IllegalArgumentException if either parameter is null
038     */
039    public MsrpInputStream(InputStream input, MsrpInputParser parser) {
040        super(input);
041        if (input == null)
042            throw new IllegalArgumentException("null input");
043        if (parser == null)
044            throw new IllegalArgumentException("null parser");
045        this.parser = parser;
046    }
047
048    /**
049     * Get the {@link MsrpInputParser} associated with this instance.
050     */
051    public MsrpInputParser getParser() {
052        return this.parser;
053    }
054
055    /**
056     * Read the next {@link MsrpMessage} from the underlying input stream.
057     *
058     * @return next message read, or null if EOF is detected
059     * @throws IOException if an I/O error occurs
060     * @throws ProtocolException if a protocol violation is detected
061     * @throws EOFException if the remote side has closed the connection in the middle of a message
062     */
063    public MsrpMessage readMsrpMessage() throws IOException {
064        while (true) {
065            final int b = this.read();
066            if (b == -1) {
067                if (!this.parser.isBetweenMessages())
068                    throw new EOFException("truncated message");
069                return null;
070            }
071            final MsrpMessage result = this.parser.inputMessageByte((byte)b);
072            if (result != null)
073                return result;
074        }
075    }
076}
