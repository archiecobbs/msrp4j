008package org.dellroad.msrp.msg;
009
010import java.io.ByteArrayOutputStream;
011import java.io.IOException;
012import java.io.OutputStream;
013import java.io.OutputStreamWriter;
014import java.util.concurrent.ThreadLocalRandom;
015import java.util.regex.Pattern;
016
017import org.dellroad.msrp.MsrpConstants;
018
019/**
020 * MSRP request/response support superclass.
021 */
022public abstract class MsrpMessage {
023
024    private final String transactionId;
025    private final MsrpHeaders headers;
026
027    /**
028     * Default constructor. Generates a random transaction ID.
029     */
030    protected MsrpMessage() {
031        this(null, null);
032    }
033
034    /**
035     * Construct an instance with the given transaction ID and headers.
036     *
037     * @param transactionId transaction ID, or null to have a randomly generated one assigned
038     * @param headers MSRP headers, or null to have an empty instance created
039     * @throws IllegalArgumentException if {@code transactionId} is invalid
040     */
041    protected MsrpMessage(String transactionId, MsrpHeaders headers) {
042        this.transactionId = transactionId != null ? transactionId : MsrpMessage.randomId();
043        if (!Pattern.compile(Util.IDENT_REGEX).matcher(this.transactionId).matches())
044            throw new IllegalArgumentException("invalid transaction ID `" + this.transactionId + "'");
045        this.headers = headers != null ? headers : new MsrpHeaders();
046    }
047
048    /**
049     * Get the transaction ID associated with this instance.
050     */
051    public String getTransactionId() {
052        return this.transactionId;
053    }
054
055    /**
056     * Get the headers associated with this instance.
057     */
058    public MsrpHeaders getHeaders() {
059        return this.headers;
060    }
061
062    /**
063     * Encode this instance according to RFC 4975.
064     *
065     * @param withBody true to include the body, or false to omit the body, if any
066     */
067    public byte[] encode(boolean withBody) {
068        try {
069            final ByteArrayOutputStream buf = new ByteArrayOutputStream();
070            final OutputStreamWriter bufWriter = new OutputStreamWriter(buf, Util.UTF8);
071            bufWriter.write(this.getFirstLine());
072            bufWriter.write(Util.CRLF);
073            bufWriter.write(this.headers.toString());
074            bufWriter.flush();
075            if (withBody)
076                this.writePayload(buf);
077            bufWriter.write(MsrpConstants.END_LINE_PREFIX);
078            bufWriter.write(this.transactionId);
079            bufWriter.write(this.getFlagByte() & 0xff);
080            bufWriter.write(Util.CRLF);
081            bufWriter.flush();
082            return buf.toByteArray();
083        } catch (IOException e) {
084            throw new RuntimeException("unexpected exception");
085        }
086    }
087
088    /**
089     * Generate a random identifier for use as transaction or message ID.
090     */
091    public static String randomId() {
092        return String.format("%016x", ThreadLocalRandom.current().nextLong());
093    }
094
095// Object
096
097    @Override
098    public boolean equals(Object obj) {
099        if (obj == this)
100            return true;
101        if (obj == null || obj.getClass() != this.getClass())
102            return false;
103        final MsrpMessage that = (MsrpMessage)obj;
104        if (!(this.transactionId != null ? this.transactionId.equals(that.transactionId) : that.transactionId == null))
105            return false;
106        if (!(this.headers != null ? this.headers.equals(that.headers) : that.headers == null))
107            return false;
108        return true;
109    }
110
111    @Override
112    public int hashCode() {
113        return (this.transactionId != null ? this.transactionId.hashCode() : 0)
114          ^ (this.headers != null ? this.headers.hashCode() : 0);
115    }
116
117    @Override
118    public String toString() {
119        return new String(this.encode(false), Util.UTF8);
120    }
121
122// Subclass methods
123
124    /**
125     * Get the end line flag byte.
126     */
127    protected abstract byte getFlagByte();
128
129    /**
130     * Get the first message line.
131     */
132    protected abstract String getFirstLine();
133
134    /**
135     * Write the message body, if any.
136     *
137     * <p>
138     * If this message has a body, this method should output CRLF, followed by the binary body content, followed by a final CRLF.
139     * If this message has no body, this method should not output anything.
140     * <p>
141     */
142    protected abstract void writePayload(OutputStream output) throws IOException;
143}
