008package org.dellroad.msrp.msg;
009
010import java.io.IOException;
011import java.io.OutputStream;
012import java.util.Arrays;
013import java.util.regex.Pattern;
014
015import org.dellroad.msrp.MsrpConstants;
016
017/**
018 * MSRP request.
019 */
020public class MsrpRequest extends MsrpMessage {
021
022    private final String method;
023
024    private byte[] body;
025    private boolean complete = true;
026    private boolean aborted;
027
028    /**
029     * Construct an instance with no body and a random transaction ID.
030     *
031     * @param method request method
032     */
033    public MsrpRequest(String method) {
034        this(null, method, null);
035    }
036
037    /**
038     * Construct an instance with no body.
039     *
040     * @param transactionId transaction ID, or null to have a randomly generated one assigned
041     * @param method request method
042     * @param headers MSRP headers, or null to have an empty instance created
043     * @throws IllegalArgumentException if {@code method} is null
044     * @throws IllegalArgumentException if {@code transactionId} is invalid
045     */
046    public MsrpRequest(String transactionId, String method, MsrpHeaders headers) {
047        this(transactionId, method, headers, null);
048    }
049
050    /**
051     * Construct an instance with a body.
052     *
053     * @param transactionId transaction ID
054     * @param method request method
055     * @param headers MSRP headers, or null to have an empty instance created
056     * @param body payload body, or null for none
057     * @throws IllegalArgumentException if {@code method} is null or invalid
058     * @throws IllegalArgumentException if {@code transactionId} is invalid
059     */
060    public MsrpRequest(String transactionId, String method, MsrpHeaders headers, byte[] body) {
061        super(transactionId, headers);
062        if (method == null)
063            throw new IllegalArgumentException("null method");
064        if (!Pattern.compile(Util.METHOD_REGEX).matcher(method).matches())
065            throw new IllegalArgumentException("invalid method `" + method + "'");
066        this.method = method;
067        this.body = body;
068    }
069
070    /**
071     * Get the MSRP request method associated with this instance.
072     */
073    public String getMethod() {
074        return this.method;
075    }
076
077    /**
078     * Get the body payload associated with this instance, if any.
079     *
080     * @return body, or null if there is none
081     */
082    public byte[] getBody() {
083        return this.body;
084    }
085    public void setBody(byte[] body) {
086        this.body = body;
087    }
088
089    /**
090     * Get the whether this message body was flagged as complete.
091     * Default is true.
092     */
093    public boolean isComplete() {
094        return this.complete;
095    }
096    public void setComplete(boolean complete) {
097        this.complete = complete;
098    }
099
100    /**
101     * Get the whether this message body was flagged as aborted.
102     * Default is false.
103     */
104    public boolean isAborted() {
105        return this.aborted;
106    }
107    public void setAborted(boolean aborted) {
108        this.aborted = aborted;
109    }
110
111    /**
112     * Build an {@link MsrpResponse} to this request, if appropriate, with the given error code and comment.
113     *
114     * @param code three digit error code
115     * @param comment error comment, or null
116     * @return valid response, or null if this request should not generate any response (i.e., {@link FailureReport#NO}).
117     * @throws IllegalArgumentException if {@code code} is not in the range 000 .. 999
118     */
119    public MsrpResponse buildResponse(int code, String comment) {
120        if (FailureReport.NO.equals(this.getHeaders().getFailureReport()))
121            return null;
122        final MsrpHeaders responseHeaders = new MsrpHeaders();
123        if (this.getMethod().equals(MsrpConstants.METHOD_SEND))
124            responseHeaders.getToPath().add(this.getHeaders().getFromPath().get(0));
125        else
126            responseHeaders.getToPath().addAll(this.getHeaders().getFromPath());
127        responseHeaders.getFromPath().add(this.getHeaders().getToPath().get(0));
128        return new MsrpResponse(this.getTransactionId(), code, comment, responseHeaders);
129    }
130
131    /**
132     * Determine if the given header name is a MIME header.
133     *
134     * @throws IllegalArgumentException if {@code name} is null
135     */
136    public static boolean isMimeHeader(String name) {
137        if (name == null)
138            throw new IllegalArgumentException("null name");
139        final String prefix = MsrpConstants.MIME_CONTENT_HEADER_PREFIX;
140        return name.regionMatches(true, 0, prefix, 0, prefix.length());
141    }
142
143// Object
144
145    @Override
146    public boolean equals(Object obj) {
147        if (obj == this)
148            return true;
149        if (!super.equals(obj))
150            return false;
151        final MsrpRequest that = (MsrpRequest)obj;
152        if (!this.method.equals(that.method))
153            return false;
154        if (!(this.body != null ? that.body != null && Arrays.equals(this.body, that.body) : that.body == null))
155            return false;
156        if (this.complete != that.complete)
157            return false;
158        if (this.aborted != that.aborted)
159            return false;
160        return true;
161    }
162
163    @Override
164    public int hashCode() {
165        return super.hashCode()
166          ^ this.method.hashCode()
167          ^ (this.body != null ? Arrays.hashCode(this.body) : 0)
168          ^ (this.complete ? 1 : 0)
169          ^ (this.aborted ? 2 : 0);
170    }
171
172// Subclass overrides
173
174    @Override
175    public byte getFlagByte() {
176        return this.aborted ? MsrpConstants.FLAG_ABORT :
177          this.complete ? MsrpConstants.FLAG_COMPLETE : MsrpConstants.FLAG_INCOMPLETE;
178    }
179
180    @Override
181    protected String getFirstLine() {
182        return "MSRP " + this.getTransactionId() + " " + this.method;
183    }
184
185    @Override
186    protected void writePayload(OutputStream output) throws IOException {
187        if (this.body != null) {
188            output.write('\r');
189            output.write('\n');
190            output.write(this.body);
191            output.write('\r');
192            output.write('\n');
193        }
194    }
195}
